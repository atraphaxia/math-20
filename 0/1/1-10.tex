\section{} Simplify the following.

\subsection{} \[\frac{7^{-3}}{7^{-16}}\]
	\phxproof{
		\phxeqf{}{$= 7^{13}$}{}
	}

\subsection{} \[\frac{3^2 + 9^{-1} - 9^0}{2^4 - 3^{-1}}\]
	\phxproof{
		\phxeqb{}{$= \frac{9 + \frac{1}{9} - 1}{16 - \frac{1}{3}}$}{}
		\phxeqs{}{$= \frac{\frac{81}{9} + \frac{1}{9} - \frac{9}{9}}{\frac{48}{3} - \frac{1}{3}}$}{}
		\phxeqs{}{$= \frac{\frac{73}{9}}{\frac{47}{3}}$}{}
		\phxeqs{}{$= \frac{73}{9} \cdot \frac{3}{47}$}{}
		\phxeqs{}{$= \frac{219}{423}$}{}
		\phxeqf{}{$= \frac{73}{141}$}{}
	}

\subsection{} \[\frac{8x^6y^3z^4}{2x^5yz^2}\]
	\phxproof{
		\phxeqf{}{$= 4xy^2z^2$}{}
	}

\subsection{} \[\frac{3x^{21}y^{12}z^{19}}{(-x^{-9}y^3z^{-3})(-9x^4y^6z^6)}\]
	\phxproof{
		\phxeqb{}{$= \frac{3x^{21}y^{12}z^{19}}{9x^{-9 + 4}y^{3 + 6}z^{-3 + 6}}$}{}
		\phxeqs{}{$= \frac{3x^{21}y^{12}z^{19}}{9x^{-5}y^9z^3}$}{}
		\phxeqf{}{$= \frac{x^{26}y^3z^{16}}{3}$}{}
	}

\subsection{} \[\frac{2{(x^3yz^2)}^2}{{(2x^2y^2z)}^3}\]
	\phxproof{
		\phxeqb{}{$= \frac{2\cancel{x^6}y^2z^4}{8\cancel{x^6}y^6z^3}$}{}
		\phxeqf{}{$= \frac{z}{4y^4}$}{}
	}

\subsection{} \[\frac{{(2a^3b^{-3}c^{-2})}^3}{{(a^3b^2c^{-1})}^{-2}}\]
	\phxproof{
		\phxeqb{}{$= \frac{8a^9b^{-9}c^{-6}}{a^{-6}b^{-4}c^2}$}{}
		\phxeqf{}{$= \frac{8a^{15}}{b^5c^8}$}{}
	}

\subsection{} \[{\left(\frac{2a^3b^{-3}c^{-1}}{3a^{-2}b^{-2}c^3}\right)}^3\]
	\phxproof{
		\phxeqb{}{$= \frac{8a^9b^{-9}c^{-3}}{27a^{-6}b^{-6}c^9}$}{}
		\phxeqf{}{$= \frac{8a^{15}}{27b^3c^{12}}$}{}
	}

\subsection{} \[{\left(\frac{a^2b^3c^{-1}}{ab^{-2}c^5}\right)}^4\]
	\phxproof{
		\phxeqb{}{$= \frac{a^8b^{12}c^{-4}}{a^4b^{-8}c^{20}}$}{}
		\phxeqf{}{$= \frac{a^4b^{20}}{c^{24}}$}{}
	}

\subsection{} \[{\left(\frac{4^0x^3y^{-2}z^{-4}}{3^2x^{-1}y^3z^{-5}}\right)}^{-3}\]
	\phxproof{
		\phxeqb{}{$= \frac{x^{-9}y^6z^{12}}{9^{-3}x^3y^{-9}z^{15}}$}{}
		\phxeqf{}{$= \frac{729y^{15}}{x^{12}z^3}$}{}
	}

\subsection{} \[{\left(\frac{18^5 \cdot 20^3x^{-1}y^2z^4}{15^9x^{-3}y^{-2}z^{-1}}\right)}^2\]
	\phxproof{
		\phxeqb{}{$= \frac{18^{10}20^6x^{-2}y^4z^8}{15^{18}x^{-6}y^{-4}z^{-2}}$}{}
		\phxeqs{}{$= \frac{2^{10}3^{10}3^{10}2^62^65^6x^{-2}y^4z^8}{3^{18}5^{18}x^{-6}y^{-4}z^{-2}}$}{}
		\phxeqs{}{$= \frac{2^{22}3^{20}5^6x^{-2}y^4z^8}{3^{18}5^{18}x^{-6}y^{-4}z^{-2}}$}{}
		\phxeqs{}{$= \frac{2^{22}3^2x^4y^8z^{10}}{5^{12}}$}{}
		\phxeqf{}{$= \frac{2^{22}9x^4y^8z^{10}}{5^{12}}$}{}
	}
