\section{} Perform the indicated operations and simplify.

\subsection{} \[\frac{3a - 3b}{c^2 + 3cd} \cdot \frac{2cd + 6d^2}{ac - bc}\]
	\phxproof{
		\phxeqb{}{$= \frac{3\cancel{(a - b)}}{c\cancel{(c + 3d)}} \cdot \frac{2d\cancel{(c + 3d)}}{c\cancel{(a - b)}}$}{}
		\phxeqf{}{$= \frac{6d}{c^2}$}{}
	}

\subsection{} \[\frac{x^3 - 8}{x^2 - 6x + 8} \div \frac{3x^3 + 6x^2 + 12x}{x^2 - 8x + 16}\]
	\phxproof{
		\phxeqb{}{$= \frac{x^3 - 8}{x^2 - 6x + 8} \cdot \frac{x^2 - 8x + 16}{3x^3 + 6x^2 + 12x}$}{}
		\phxeqs{}{$= \frac{\cancel{(x - 2)}\cancel{(x^2 + 2x + 4)}}{\cancel{(x - 4)}\cancel{(x - 2)}} \cdot \frac{{(x - 4)}\cancel{^2}}{3x\cancel{(x^2 + 2x + 4)}}$}{Factor.}
		\phxeqf{}{$= \frac{x - 4}{3x}$}{}
	}

\subsection{} \[\frac{x + 1}{x} - \frac{y + 1}{y}\]
	\phxproof{
		\phxeqb{}{$= \frac{xy + y}{xy} - \frac{xy + x}{xy}$}{$\lcm = xy$}
		\phxeqf{}{$= \frac{y - x}{xy}$}{}
	}

\subsection{} \[\frac{a}{bc} + \frac{b}{ac} - \frac{c}{ab}\]
	\phxproof{
		\phxeqb{}{$= \frac{a^2}{abc} + \frac{b^2}{abc} - \frac{c^2}{abc}$}{$\lcm = abc$}
		\phxeqb{}{$= \frac{a^2 + b^2 - c^2}{abc}$}{}
	}

\subsection{} \[\frac{2}{x^2} + \frac{3}{xy} - \frac{1}{y^2}\]
	\phxproof{
		\phxeqb{}{$= \frac{2y^2}{x^2y^2} + \frac{3xy}{x^2y^2} - \frac{x^2}{x^2y^2}$}{$\lcm = x^2y^2$}
		\phxeqf{}{$= \frac{2y^2 + 3xy - x^2}{x^2y^2}$}{}
	}

\subsection{} \[\frac{1}{x} - \frac{2}{x + y}\]
	\phxproof{
		\phxeqb{}{$= \frac{x + y}{x(x + y)} - \frac{2x}{x(x + y)}$}{$\lcm = x(x + y)$}
		\phxeqf{}{$= \frac{y - x}{x(x + y)}$}{}
	}

\subsection{} \[\frac{2}{x - y} + \frac{3}{y - x}\]
	\phxproof{
		\phxeqb{}{$= \frac{3}{y - x} - \frac{2}{y - x}$}{$\frac{2}{x - y} = \frac{2}{-(y - x)} = -\frac{2}{y - x}$}
		\phxeqf{}{$= \frac{1}{y - x}$}{}
	}

\subsection{} \[\frac{2}{(a + b)(a - b)} + \frac{1}{{(a - b)}^2}\]
	\phxproof{
		\phxeqb{}{$= \frac{2a - 2b}{(a + b){(a - b)}^2} + \frac{a + b}{(a + b){(a - b)}^2}$}{$\lcm = (a + b){(a - b)}^2$}
		\phxeqf{}{$= \frac{3a - b}{(a + b){(a - b)}^2}$}{}
	}

\subsection{} \[\frac{x^2}{x(x - 1)} + \frac{1}{x} - \frac{1}{x - 1}\]
	\phxproof{
		\phxeqb{}{$= \frac{x^2}{x(x - 1)} + \frac{x - 1}{x(x - 1)} - \frac{x}{x(x - 1)}$}{$\lcm = x(x - 1)$}
		\phxeqs{}{$= \frac{x^2 - 1}{x(x - 1)}$}{}
		\phxeqs{}{$= \frac{\cancel{(x - 1)}(x + 1)}{x\cancel{(x - 1)}}$}{Factor using difference of two squares.}
		\phxeqf{}{$= \frac{x + 1}{x}$}{}
	}

\subsection{} \[\frac{4}{4x^2 - 9} - \frac{3}{2x^2 - x - 3}\]
	\phxproof{
		\phxeqb{}{$= \frac{4}{(2x - 3)(2x + 3)} - \frac{3}{2x^2 - 3x + 2x - 3}$}{Factor.}
		\phxeqs{}{$= \frac{4}{(2x - 3)(2x + 3)} - \frac{3}{(x + 1)(2x - 3)}$}{}
		\phxeqs{}{$= \frac{4x + 4}{(x + 1)(2x - 3)(2x + 3)} - \frac{6x + 9}{(x + 1)(2x - 3)(2x + 3)}$}{$\lcm = (x + 1)(2x - 3)(2x + 3)$}
		\phxeqf{}{$= \frac{-2x - 5}{(x + 1)(2x - 3)(2x + 3)}$}{}
	}
