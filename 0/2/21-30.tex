\subsection{} \[2r^2 - r - 3\]
	\phxproof{
		\phxeqb{}{$= 2r^2 + 2r - 3r - 3$}{Factor by grouping.}
		\phxeqf{}{$= (2r - 3)(r + 1)$}{}
	}

\subsection{} \[9a^2 + 15ab + 4b^2\]
	\phxproof{
		\phxeqb{}{$= 9a^2 + 12ab + 3ab + 4b^2$}{Factor by grouping.}
		\phxeqf{}{$= (3a + b)(3a + 4b)$}{}
	}

\subsection{} \[7s^6 - 9s^3 + 2\]
	\phxproof{
		\phxeqb{}{$= 7s^6 - 7s^3 - 2s^3 + 2$}{Factor by grouping.}
		\phxeqf{}{$= (7s^3 - 2)(s^3 - 1)$}{}
	}

\subsection{} \[a^2 + ab + 5a + 5b\]
	\phxproof{
		\phxeqb{}{$= a(a + b) + 5(a + b)$}{}
		\phxeqf{}{$= (a + 5)(a + b)$}{}
	}

\subsection{} \[x^3 - x^2 - x + 1\]
	\phxproof{
		\phxeqb{}{$= x^2(x - 1) - (x - 1)$}{}
		\phxeqs{}{$= (x^2 - 1)(x - 1)$}{}
		\phxeqf{}{$= {(x - 1)}^2(x + 1)$}{Factor using difference of two squares.}
	}
