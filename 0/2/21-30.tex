\subsection{} \[2r^2 - r - 3\]
	\phxproof{
		\phxeqb{}{$= 2r^2 + 2r - 3r - 3$}{Factor by grouping.}
		\phxeqf{}{$= (2r - 3)(r + 1)$}{}
	}

\subsection{} \[9a^2 + 15ab + 4b^2\]
	\phxproof{
		\phxeqb{}{$= 9a^2 + 12ab + 3ab + 4b^2$}{Factor by grouping.}
		\phxeqf{}{$= (3a + b)(3a + 4b)$}{}
	}

\subsection{} \[7s^6 - 9s^3 + 2\]
	\phxproof{
		\phxeqb{}{$= 7s^6 - 7s^3 - 2s^3 + 2$}{Factor by grouping.}
		\phxeqf{}{$= (7s^3 - 2)(s^3 - 1)$}{}
	}

\subsection{} \[a^2 + ab + 5a + 5b\]
	\phxproof{
		\phxeqb{}{$= a(a + b) + 5(a + b)$}{}
		\phxeqf{}{$= (a + 5)(a + b)$}{}
	}

\subsection{} \[x^3 - x^2 - x + 1\]
	\phxproof{
		\phxeqb{}{$= x^2(x - 1) - (x - 1)$}{}
		\phxeqs{}{$= (x^2 - 1)(x - 1)$}{}
		\phxeqf{}{$= {(x - 1)}^2(x + 1)$}{Factor using difference of two squares.}
	}

\subsection{} \[64x^4 + 1\]
	\phxproof{
		\phxeqb{}{$= 64x^4 + 16x^2 + 1 - 16x^2$}{Find $b$ such that the expression becomes a perfect square trinomial.}
		\phxeqs{}{$= {(8x^2 + 1)}^2 - 16x^2$}{}
		\phxeqf{}{$= (8x^2 - 4x + 1)(8x^2 + 4x + 1)$}{Factor using difference of two squares.}
	}

\subsection{} \[4m^8 - 4m^4 + 9\]
	\phxproof{
		\phxeqb{}{$= 4m^8 + 12m^4 + 9 - 16m^4$}{Find $b$ such that the expression becomes a perfect square trinomial.}
		\phxeqs{}{$= {(2m^4 + 3)}^2 - 16m^4$}{}
		\phxeqs{}{$= (2m^4 - 4m^2 + 3)(2m^4 + 4m^2 + 3)$}{Factor using difference of two squares.}
	}
