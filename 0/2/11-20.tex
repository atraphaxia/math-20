\subsection{} \[8j^3 - 125k^6\]
	\phxproof{
		\phxeqf{}{$= (2j - 5k^2)(4j^2 + 10jk^2 + 25k^4)$}{Factor using difference of two cubes.}
	}

\subsection{} \[{(x - y)}^3 - 1\]
	\phxproof{
		\phxeqf{}{$= (x - y - 1)({(x - y)}^2 + x - y + 1)$}{Factor using difference of two cubes.}
	}

\subsection{} \[x^2 - 2x - 3\]
	\phxproof{
		\phxeqf{}{$= (x - 3)(x + 1)$}{Factor by grouping.}
	}

\subsection{} \[s^2 + 7s + 10\]
	\phxproof{
		\phxeqf{}{$= (x + 5)(x + 2)$}{Factor by grouping.}
	}

\subsection{} \[48 - 13q - q^2\]
	\phxproof{
		\phxeqb{}{$= -(q^2 + 13q - 48)$}{}
		\phxeqf{}{$= -(q + 16)(q - 3)$}{Factor by grouping.}
	}

\subsection{} \[z^6 - 6z^3 - 72\]
	\phxproof{
		\phxeqf{}{$= (z^3 - 12)(z^3 + 6)$}{Factor by grouping.}
	}

\subsection{} \[t^4 - 9st^2 - 10s^2\]
	\phxproof{
		\phxeqb{}{$= t^4 + st^2 - 10st^2 - 10s^2$}{Factor by grouping.}
		\phxeqf{}{$= (t^2 - 10s)(t^2 + s)$}{}
	}

\subsection{} \[a^6b^4 - a^3b^2c - 42c^2\]
	\phxproof{
		\phxeqb{}{$= a^6b^4 + 6a^3b^2c - 7a^3b^2c - 42c^2$}{Factor by grouping.}
		\phxeqf{}{$= (a^3b^2 - 7c)(a^3b^2 + 6c)$}{}
	}

\subsection{} \[4x^2 - 4x + 1\]
	\phxproof{
		\phxeqb{}{$= 4x^2 - 2x - 2x + 1$}{Factor by grouping.}
		\phxeqf{}{$= {(2x - 1)}^2$}{}
	}

\subsection{} \[4n^2 - 12n + 9\]
	\phxproof{
		\phxeqb{}{$= 4n^2 - 6n - 6n + 9$}{Factor by grouping.}
		\phxeqf{}{$= {(2n - 3)}^2$}{}
	}
