\section{} Factor completely.

\subsection{} \[4x^3y^2 + 6x^2y^3\]
	\phxproof{
		\phxeqf{}{$= 2x^2y^2(2x + 3y)$}{}
	}

\subsection{} \[a^2b^3c^4 - a^3b^4c^5 + 2a^2b^4c^4\]
	\phxproof{
		\phxeqf{}{$= a^2b^3c^4(1 - abc + 2b)$}{}
	}

\subsection{} \[9a^2 - 1\]
	\phxproof{
		\phxeqf{}{$= (3a - 1)(3a + 1)$}{Factor using difference of two squares.}
	}

\subsection{} \[16x^2 - 25y^2\]
	\phxproof{
		\phxeqf{}{$= (4x - 5y)(4x + 5y)$}{Factor using difference of two squares.}
	}

\subsection{} \[25x^4y^8 - 64z^2\]
	\phxproof{
		\phxeqf{}{$= (5x^2y^4 - 8z)(5x^2y^4 + 8z)$}{Factor using difference of two squares.}
	}

\subsection{} \[16x^4 - 1\]
	\phxproof{
		\phxeqb{}{$= (4x^2 - 1)(4x^2 + 1)$}{Factor using difference of two squares.}
		\phxeqf{}{$= (2x - 1)(2x + 1)(4x^2 + 1)$}{Factor using difference of two squares.}
	}

\subsection{} \[{(s - 2t)}^2 - 4\]
	\phxproof{
		\phxeqf{}{$= (s - 2t - 2)(s - 2t + 2)$}{Factor using difference of two squares.}
	}

\subsection{} \[{(x + 3)}^2 - {(y + 2)}^2\]
	\phxproof{
		\phxeqb{}{$= (x + 3 - y - 2)(x + 3 + y + 2)$}{Factor using difference of two squares.}
		\phxeqf{}{$= (x - y + 1)(x + y + 5)$}{}
	}

\subsection{} \[x^2(x^2 - 1) - 9(x^2 - 1)\]
	\phxproof{
		\phxeqb{}{$= (x^2 - 9)(x^2 - 1)$}{}
		\phxeqf{}{$= (x - 3)(x + 3)(x - 1)(x + 1)$}{Factor using difference of two squares.}
	}

\subsection{} \[8x^9 + z^{12}\]
	\phxproof{
		\phxeqf{}{$= (2x^3 + z^4)(4x^6 - 2x^3z^4 + z^8)$}{Factor using sum of two cubes.}
	}
