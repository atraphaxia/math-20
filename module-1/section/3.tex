\section{}
	Find the solution set of the following inequalities.

	\subsection{}
		\[x - 2 < 4 - x\]
		\begin{align*}
			2x &< 6 \\
			x &< 3 \\
			\Aboxed{x &\in (-\infty, 3)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[x + 1 > \frac{2x - 4}{5}\]
		\begin{align*}
			5x + 5 &> 2x - 4 \\
			3x &> -9 \\
			x &> -3 \\
			\Aboxed{x &\in (-3, +\infty)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{2x - 3}{4} \geq \frac{7x}{2}\]
		\begin{align*}
			2x - 3 &\geq 14x \\
			12x &\leq -3 \\
			x &\leq -\frac{1}{4} \\
			\Aboxed{x &\in \left(-\infty, -\frac{1}{4}\right]}
		\end{align*}
		\hrule{}

	\subsection{}
		\[-2 < 5 + 3x < 20\]

			-7 < 3x &< 15 \\
			-\frac{7}{3} < x &< 5 \\
			\Aboxed{x &\in \left(-\frac{7}{3}, 5\right)}
		\end{align*}
		\hrule{}

		\end{align*}

	\subsection{}
		\[2x - 1 \leq 3x \leq 4x + 1\]
