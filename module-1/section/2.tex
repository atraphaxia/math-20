\section{}
	Find all real values of $k$ such that there is exactly one solution to the equation
	\[x^2 - kx + k = x - 2\]
	\begin{align*}
		\intertext{Recall that a quadratic equation has exactly one solution if and only if
		its discriminant $b^2 - 4ac$ is equal to $0$. Rewrite the equation in standard form.}
		x^2 - kx - x + k + 2 &= 0 \\
		x^2 - (k + 1)x + (k + 2) &= 0 \\
		\intertext{Substitute into the discriminant.}
		b^2 - 4ac &= 0 \\
		{(k + 1)}^2 - 4(1)(k + 2) &= 0 \\
		k^2 + 2k + 1 - 4k - 8 &= 0 \\
		k^2 - 2k - 7 &= 0 \\
		k &= \frac{2 \pm \sqrt{{(-2)}^2 - 4(1)(-7)}}{2(1)} \\
		k &= \frac{2 \pm \sqrt{4 + 28}}{2} \\
		k &= \frac{2 \pm 4\sqrt{2}}{2} \\
		\Aboxed{k &= 1 \pm 2\sqrt{2}} \\
	\end{align*}
