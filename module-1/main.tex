	\section{}
		Solve the following equations.

		\subsection{}
			\[x + 9 = 5 - 3x\]
			\begin{align*}
				4x &= -4 \\
				\Aboxed{x &= -1}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{2x + 3}{4} - \frac{x + 1}{2} = -\frac{1}{3}\]
			\begin{align*}
				6x + 9 - 6x - 6 &= -4 \\
				3 &= -4 \\
				\Aboxed{x &\in \emptyset}
			\end{align*}
			\hrule{}

		\subsection{}
			\[3x^2 - 2x + 1 = 0\]
			\begin{align*}
				x &= \frac{2 \pm \sqrt{{(-2)}^2 - 4(3)(1)}}{2(3)} \\
				x &= \frac{2 \pm \sqrt{4 - 12}}{6} \\
				x &= \frac{2 \pm \sqrt{8}i}{6} \\
				\Aboxed{x &= \frac{1 \pm \sqrt{2}i}{3}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[4x^2 + 2x = 2\]
			\begin{align*}
				2x^2 + x &= 1 \\
				2x^2 + x - 1 &= 0 \\
				2x^2 + 2x - x - 1 &= 0 \\
				(2x - 1)(x + 1) &= 0 \\
				\Aboxed{x &\in \left\{-1, \frac{1}{2}\right\}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[16x^2 + 9 = 24x\]
			\begin{align*}
				16x^2 - 24x + 9 &= 0 \\
				16x^2 - 12x - 12x + 9 &= 0 \\
				{(4x - 3)}^2 &= 0 \\
				\Aboxed{x &= \frac{3}{4}}
			\end{align*}

		\subsection{}
			\[\frac{x}{x - 1} + \frac{x - 5}{x^2 + 2x - 3} = \frac{1}{x + 3}\]
			\begin{align*}
				\frac{x}{x - 1} + \frac{x - 5}{(x + 3)(x - 1)} - \frac{1}{x + 3} &= 0 \\
				\intertext{$x \notin \{-3, 1\}$}
				x^2 + 3x + x - 5 - x + 1 &= 0 \\
				x^2 + 3x - 4 &= 0 \\
				(x + 4)(x - 1) &= 0 \\
				\Aboxed{x &= 4}
 			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{1}{4 - x} + \frac{3}{6 + x} = 0\]
			\begin{align*}
				\intertext{$x \notin \{-6, 4\}$}
				6 + x + 12 - 3x &= 0 \\
				18 &= 2x \\
				\Aboxed{x &= 9}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3}{x^2 - 9} - \frac{7}{x - 3} = -\frac{4}{x + 3}\]
			\begin{align*}
				\frac{3}{(x - 3)(x + 3)} - \frac{7}{x - 3} + \frac{4}{x + 3} &= 0 \\
				\intertext{$x \notin \{-3, 3\}$}
				3 - 7x - 21 + 4x - 12 &= 0 \\
				-30 &= 3x \\
				\Aboxed{x &= -10}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3}{x^2 - x - 6} = \frac{4}{2x^2 + x - 6}\]
			\begin{align*}
				\frac{3}{(x - 3)(x + 2)} &= \frac{4}{2x^2 + 4x - 3x - 6} \\
				\frac{3}{(x - 3)(x + 2)} - \frac{4}{(2x - 3)(x + 2)} &= 0 \\
				\intertext{$x \notin \left\{-2, \frac{3}{2}, 3\right\}$} \\
				6x - 9 - 4x + 12 &= 0 \\
				2x &= -3 \\
				\Aboxed{x &= -\frac{3}{2}}
			\end{align*}

		\subsection{}
			\[\frac{2}{y + 1} - \frac{3}{1 - y} = \frac{5}{y}\]
			\begin{align*}
				\frac{2}{y + 1} - \frac{3}{1 - y} - \frac{5}{y} &= 0 \\
				\intertext{$y \notin \{-1, 0, 1\}$}
				2y(1 - y) - 3y(y + 1) - 5(y + 1)(1 - y) &= 0 \\
				2y - 2y^2 - 3y^2 - 3y - 5(y - y^2 + 1 - y) &= 0 \\
				-5y^2 - y + 5y^2 - 5 &= 0 \\
				\Aboxed{y &= -5}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{x + 17}{x^2 - 6x + 8} + \frac{x - 2}{x - 4} = \frac{x - 4}{x - 2}\]
			\begin{align*}
				\frac{x + 17}{(x - 4)(x - 2)} + \frac{x - 2}{x - 4} - \frac{x - 4}{x - 2} &= 0 \\
				\intertext{$x \notin \{2, 4\}$}
				x + 17 + x^2 - 2x + 4 - x^2 + 8x - 16 &= 0 \\
				5x &= -5 \\
				\Aboxed{x &= -1}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{w}{3w^2 - 8w + 4} = \frac{w + 2}{3w^2 + w - 2}\]
			\begin{align*}
				\frac{w}{3w^2 - 6w - 2w + 4} - \frac{w + 2}{3w^2 + 3w - 2w - 2} &= 0 \\
				\frac{w}{(3w - 2)(w - 2)} - \frac{w + 2}{(3w - 2)(w + 1)} &= 0 \\
				\intertext{$w \notin \left\{-1, \frac{2}{3}, 2\right\}$}
				w^2 + w - w^2 + 4 &= 0 \\
				\Aboxed{w &= -4}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3}{x + 4} = \frac{2}{3x - 2}\]
			\begin{align*}
				\frac{3}{x + 4} - \frac{2}{3x - 2} &= 0 \\
				\intertext{$x \notin \left\{-4, \frac{2}{3}\right\}$}
				9x - 6 - 2x - 8 &= 0 \\
				7x &= 14 \\
				\Aboxed{x &= 2}
			\end{align*}

		\subsection{}
			\[\frac{x + 4}{x + 2} = \frac{2x - 3}{x - 5}\]
			\begin{align*}
				\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} &= 0 \\
				\intertext{$x \notin \{-2, 5\}$}
				(x + 4)(x - 5) - (2x - 3)(x + 2) &= 0 \\
				x^2 - 5x + 4x - 20 - 2x^2 - 4x + 3x + 6 &= 0 \\
				-x^2 - 2x - 14 &= 0 \\
				x^2 + 2x + 14 &= 0 \\
				x &= \frac{-2 \pm \sqrt{2^2 - 4(1)(14)}}{2(1)} \\
				x &= \frac{-2 \pm \sqrt{4 - 56}}{2} \\
				x &= \frac{-2 \pm \sqrt{-52}}{2} \\
				x &= \frac{-2 \pm 2\sqrt{13}i}{2} \\
				\Aboxed{x &= -1 \pm \sqrt{13}i}
			\end{align*}
