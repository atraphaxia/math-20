			\end{align*}

			\[3x^2 - 2x + 1 = 0\]

				x &= \frac{2 \pm \sqrt{{(-2)}^2 - 4(3)(1)}}{2(3)} \\
				x &= \frac{2 \pm \sqrt{4 - 12}}{6} \\
				x &= \frac{2 \pm 2i\sqrt{2}}{6} \\
				\Aboxed{x &= \frac{1 \pm i\sqrt{2}}{3}}
			\end{align*}

			\[4x^2 + 2x = 2\]

				\Aboxed{x &\in \left\{-1, \frac{1}{2}\right\}}
			\end{align*}

			\[16x^2 + 9 = 24x\]

			\end{align*}

		\subsection{}
			\[\frac{x}{x - 1} + \frac{x - 5}{x^2 + 2x - 3} = \frac{1}{x + 3}\]
			\begin{align*}
				\frac{x}{x - 1} + \frac{x - 5}{(x - 1)(x + 3)} &= \frac{1}{x + 3} \\
				\intertext{$x \notin \{-3, 1\}$}
				x^2 + 3x + x - 5 &= x - 1 \\
				x^2 + 3x - 4 &= 0 \\
				(x + 4)(x - 1) &= 0 \\
				x &\in \{-4, 1\} \\
				\Aboxed{x &= -4}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{1}{4 - x} + \frac{3}{6 + x} = 0\]
			\begin{align*}
				\intertext{$x \notin \{-6, 4\}$}

				2x &= 18 \\
				\Aboxed{x &= 9}
			\end{align*}

		\subsection{}
			\[\frac{3}{x^2 - 9} - \frac{7}{x - 3} = -\frac{4}{x + 3}\]

				\intertext{$x \notin \{-3, 3\}$}
				3 - 7x - 21 &= -4x + 12 \\
				3x &= -30 \\
				\Aboxed{x &= -10}
			\end{align*}

			\[\frac{3}{x^2 - x - 6} = \frac{4}{2x^2 + x - 6}\]

				\frac{3}{(x - 3)(x + 2)} &= \frac{4}{2x^2 + 4x - 3x - 6} \\
				\frac{3}{(x - 3)(x + 2)} &= \frac{4}{(2x - 3)(x + 2)} \\
				\intertext{$x \notin \left\{-2, \frac{3}{2}, 3\right\}$}
				6x - 9 &= 4x - 12 \\

				x - 1 &= x - 2 - 2\sqrt{x - 2} + 1 \\

				4x - 8 &= 0 \\
				4x &= 8 \\
				x &= 2 \\

			\[2\sqrt{x - 1} - \sqrt{x - 2} - 2 = 0\]

				9x^2 - 52x + 68 &= 0 \\
				9x^2 - 18x - 34x + 68 &= 0 \\

				2\sqrt{\frac{25}{9}} - \sqrt{\frac{16}{9}} - 2 &= 0 \\
				\frac{10}{3} - \frac{4}{3} - 2 &= 0 \\
				0 &= 0 \\
				\Aboxed{x &\in \left\{2, \frac{34}{9}\right\}}
			\end{align*}

			\[\sqrt{3 - 3x} - \sqrt{3x + 2} = 3\]

				9x^2 - 3x - 2 &= 0 \\
				9x^2 - 6x + 3x - 2 &= 0 \\
				(3x + 1)(3x - 2) &= 0 \\
				x &\in \left\{-\frac{1}{3}, \frac{2}{3}\right\} \\

				1 &\neq 3 \\
				\intertext{Check $x = \frac{2}{3}$.}

				-1 &\neq 3 \\

		\subsection{}
			\[2 + 2\sqrt[3]{x - 1} = x\]
			\begin{align*}
				2\sqrt[3]{x - 1} &= x - 2 \\
				8x - 8 &= x^3 - 6x^2 + 12x - 8 \\
				x^3 - 6x^2 + 4x &= 0 \\
				x(x^2 - 6x + 4) &= 0 \\
				\intertext{$x = 0$ may be a solution. Solve for the zeroes of $x^2 - 6x + 4$.}
				x &= \frac{6 \pm \sqrt{{(-6)}^2 - 4(1)(4)}}{2(1)} \\
				x &= \frac{6 \pm \sqrt{36 - 16}}{2} \\
				x &= \frac{6 \pm 2\sqrt{5}}{2} \\
				x &= 3 \pm \sqrt{5} \\
				x &\in \{0, 3 \pm \sqrt{5}\} \\
				\intertext{Check $x = 0$.}
				2 + 2\sqrt[3]{0 - 1} &= 0 \\
				0 &= 0 \\
				\intertext{Check $x = 3 + \sqrt{5}$.}
				2 + 2\sqrt[3]{3 + \sqrt{5} - 1} &= 3 + \sqrt{5} \\
				2\sqrt[3]{2 + \sqrt{5}} &= 1 + \sqrt{5} \\
				16 + 8\sqrt{5} &= 1 + 3\sqrt{5} + 15 + 5\sqrt{5} \\
				16 + 8\sqrt{5} &= 16 + 8\sqrt{5} \\
				\intertext{Check $x = 3 - \sqrt{5}$.}
				2 + 2\sqrt[3]{3 - \sqrt{5} - 1} &= 3 - \sqrt{5} \\
				2\sqrt[3]{2 - \sqrt{5}} &= 1 - \sqrt{5} \\
				16 - 8\sqrt{5} &= 1 - 3\sqrt{5} + 15 - 5\sqrt{5} \\
				16 - 8\sqrt{5} &= 16 - 8\sqrt{5} \\
				\Aboxed{x &\in \{0, 3 \pm \sqrt{5}\}}
			\end{align*}
			\pagebreak{}

		\subsection{}
			\[\sqrt[4]{3x^2 - 3x + 1} = \sqrt{x - 2}\]
			\begin{align*}
				\sqrt{3x^2 - 3x + 1} &= x - 2 \\
				3x^2 - 3x + 1 &= x^2 - 4x + 4 \\
				2x^2 + x - 3 &= 0 \\
				2x^2 - 2x + 3x - 3 &= 0 \\
				(2x + 3)(x - 1) &= 0 \\
				x &\in \left\{-\frac{3}{2}, 1\right\} \\
				\intertext{Check $x = -\frac{3}{2}$.}
				\sqrt[4]{3{\left(-\frac{3}{2}\right)}^2 - 3\left(-\frac{3}{2}\right) + 1} &= \sqrt{-\frac{3}{2} - 2} \\
				\sqrt[4]{3\left(\frac{9}{4}\right) - 3\left(-\frac{6}{4}\right) + \frac{4}{4}} &= \sqrt{-\frac{3}{2} - \frac{4}{2}} \\
				\sqrt[4]{\frac{27}{4} + \frac{18}{4} + \frac{4}{4}} &= i\sqrt{\frac{7}{2}} \\
				\sqrt[4]{\frac{49}{4}} &= i\sqrt{\frac{7}{2}} \\
				\sqrt{\frac{7}{2}} &\neq i\sqrt{\frac{7}{2}} \\
				\intertext{Check $x = 1$.}
				\sqrt[4]{3{(1)}^2 - 3(1) + 1} &= \sqrt{1 - 2} \\
				1 &\neq i \\
				\Aboxed{x &\in \emptyset}
			\end{align*}
			\pagebreak{}

		\subsection{}
			\[\sqrt{2x + 3} - \sqrt{x - 2} = \sqrt{x + 1}\]
			\begin{align*}
				\sqrt{2x + 3} &= \sqrt{x + 1} + \sqrt{x - 2} \\
				2x + 3 &= x + 1 + 2\sqrt{x + 1}\sqrt{x - 2} + x - 2 \\
				4 &= 2\sqrt{x + 1}\sqrt{x - 2} \\
				16 &= 4(x + 1)(x - 2) \\
				16 &= 4(x^2 - x - 2) \\
				16 &= 4x^2 - 4x - 8 \\
				4x^2 - 4x - 24 &= 0 \\
				x^2 - x - 6 &= 0 \\
				(x - 3)(x + 2) &= 0 \\
				x &\in \{-2, 3\} \\
				\intertext{Check $x = -2$.}
				\sqrt{2(-2) + 3} - \sqrt{-2 - 2} &= \sqrt{-2 + 1} \\
				-i &\neq i \\
				\intertext{Check $x = 3$.}
				\sqrt{2(3) + 3} - \sqrt{3 - 2} &= \sqrt{3 + 1} \\
				2 &= 2 \\
				\Aboxed{x &= 3}
			\end{align*}
			\hrule{}

		\subsection{}
			\[1 = x + \sqrt{2x - 3}\]
			\begin{align*}
				1 - x &= \sqrt{2x - 3} \\
				1 - 2x + x^2 &= 2x - 3 \\
				x^2 - 4x + 4 &= 0 \\
				{(x - 2)}^2 &= 0 \\
				x &= 2 \\
				\intertext{Check $x = 2$.}
				1 &= 2 + \sqrt{2(2) - 3} \\
				1 &\neq 3 \\
				\Aboxed{x &\in \emptyset}
			\end{align*}
			\pagebreak{}
