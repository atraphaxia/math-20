	\section{}
		Solve the following equations.

		\subsection{}
			\[x + 9 = 5 - 3x\]
			\begin{align*}
				4x &= -4 \\
				\Aboxed{x &= -1}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{2x + 3}{4} - \frac{x + 1}{2} = -\frac{1}{3}\]
			\begin{align*}
				6x + 9 - 6x - 6 &= -4 \\
				3 &\neq -4 \\
				\Aboxed{x &\in \emptyset}
			\end{align*}
			\hrule{}

		\subsection{}
			\[3x^2 - 2x + 1 = 0\]
			\begin{align*}
				x &= \frac{2 \pm \sqrt{{(-2)}^2 - 4(3)(1)}}{2(3)} \\
				x &= \frac{2 \pm \sqrt{4 - 12}}{6} \\
				x &= \frac{2 \pm 2\sqrt{2}i}{6} \\
				\Aboxed{x &= \frac{1 \pm \sqrt{2}i}{3}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[4x^2 + 2x = 2\]
			\begin{align*}
				4x^2 + 2x - 2 &= 0 \\
				2x^2 + x - 1 &= 0 \\
				2x^2 + 2x - x - 1 &= 0 \\
				(2x - 1)(x + 1) &= 0 \\
				\Aboxed{x &\in \left\{-1, \frac{1}{2}\right\}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[16x^2 + 9 = 24x\]
			\begin{align*}
				16x^2 - 24x + 9 &= 0 \\
				{(4x - 3)}^2 &= 0 \\
				\Aboxed{x &= \frac{3}{4}}
			\end{align*}

		\subsection{}
			\[\frac{x}{x - 1} + \frac{x - 5}{x^2 + 2x - 3} = \frac{1}{x + 3}\]
			\begin{align*}
				\frac{x}{x - 1} + \frac{x - 5}{(x - 1)(x + 3)} &= \frac{1}{x + 3} \\
				\intertext{$x \notin \{-3, 1\}$}
				x^2 + 3x + x - 5 &= x - 1 \\
				x^2 + 3x - 4 &= 0 \\
				(x + 4)(x - 1) &= 0 \\
				x &\in \{-4, 1\} \\
				\Aboxed{x &= -4}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{1}{4 - x} + \frac{3}{6 + x} = 0\]
			\begin{align*}
				\intertext{$x \notin \{-6, 4\}$}
				6 + x + 12 - 3x &= 0 \\
				2x &= 18 \\
				\Aboxed{x &= 9}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3}{x^2 - 9} - \frac{7}{x - 3} = -\frac{4}{x + 3}\]
			\begin{align*}
				\frac{3}{(x - 3)(x + 3)} - \frac{7}{x - 3} &= -\frac{4}{x + 3} \\
				\intertext{$x \notin \{-3, 3\}$}
				3 - 7x - 21 &= -4x + 12 \\
				3x &= -30 \\
				\Aboxed{x &= -10}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3}{x^2 - x - 6} = \frac{4}{2x^2 + x - 6}\]
			\begin{align*}
				\frac{3}{(x - 3)(x + 2)} &= \frac{4}{2x^2 + 4x - 3x - 6} \\
				\frac{3}{(x - 3)(x + 2)} &= \frac{4}{(2x - 3)(x + 2)} \\
				\intertext{$x \notin \left\{-2, \frac{3}{2}, 3\right\}$}
				6x - 9 &= 4x - 12 \\
				2x &= -3 \\
				\Aboxed{x &= -\frac{3}{2}}
			\end{align*}

		\subsection{}
			\[\frac{2}{y + 1} - \frac{3}{1 - y} = \frac{5}{y}\]
			\begin{align*}
				\intertext{$y \notin \{-1, 0, 1\}$}
				2y(1 - y) - 3y(y + 1) &= 5(y + 1)(1 - y) \\
				2y - 2y^2 - 3y^2 - 3y &= 5(y - y^2 + 1 - y) \\
				-5y^2 - y &= -5y^2 + 5 \\
				\Aboxed{y &= -5} \\
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{x + 17}{x^2 - 6x + 8} + \frac{x - 2}{x - 4} = \frac{x - 4}{x - 2}\]
			\begin{align*}
				\frac{x + 17}{(x - 4)(x - 2)} + \frac{x - 2}{x - 4} &= \frac{x - 4}{x - 2} \\
				\intertext{$x \notin \{2, 4\}$}
				x + 17 + x^2 - 4x + 4 &= x^2 - 8x + 16 \\
				5x &= -5 \\
				\Aboxed{x &= -1}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{w}{3w^2 - 8w + 4} = \frac{w + 2}{3w^2 + w - 2}\]
			\begin{align*}
				\frac{w}{3w^2 - 6w - 2w + 4} &= \frac{w + 2}{3w^2 + 3w - 2w - 2} \\
				\frac{w}{(3w - 2)(w - 2)} &= \frac{w + 2}{(3w - 2)(w + 1)} \\
				\intertext{$w \notin \left\{-1, \frac{2}{3}, 2\right\}$}
				w^2 + w &= w^2 - 4 \\
				\Aboxed{w &= -4}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3}{x + 4} = \frac{2}{3x - 2}\]
			\begin{align*}
				\intertext{$x \notin \left\{-4, \frac{2}{3}\right\}$}
				9x - 6 &= 2x + 8 \\
				7x &= 14 \\
				\Aboxed{x &= 2}
			\end{align*}

		\subsection{}
			\[\frac{x + 4}{x + 2} = \frac{2x - 3}{x - 5}\]
			\begin{align*}
				\intertext{$x \notin \{-2, 5\}$}
				(x + 4)(x - 5) &= (2x - 3)(x + 2) \\
				x^2 - 5x + 4x - 20 &= 2x^2 + 4x - 3x - 6 \\
				x^2 + 2x + 14 &= 0 \\
				x &= \frac{-2 \pm \sqrt{2^2 - 4(1)(14)}}{2(1)} \\
				x &= \frac{-2 \pm i\sqrt{52}}{2} \\
				x &= \frac{-2 \pm 2i\sqrt{13}}{2} \\
				\Aboxed{x &= -1 \pm i\sqrt{13}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{4}{z - 2} = \frac{z}{6}\]
			\begin{align*}
				\intertext{$z \neq 2$}
				24 &= z^2 - 2z \\
				z^2 - 2z - 24 &= 0 \\
				(z - 6)(z + 4) &= 0 \\
				\Aboxed{z &\in \{-4, 6\}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{1}{2x + 5} = \frac{4}{2x - 1}\]
			\begin{align*}
				\intertext{$x \notin \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}
				2x - 1 &= 8x + 20 \\
				6x &= -21 \\
				x &= -\frac{21}{6} \\
				\Aboxed{x &= -\frac{7}{2}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{2}{3x} + \frac{4}{5x} = \frac{11}{15}\]
			\begin{align*}
				\intertext{$x \neq 0$}
				10 + 12 &= 11x \\
				\Aboxed{x &= 2}
			\end{align*}

		\subsection{}
			\[\frac{5}{x + 3} = \frac{x + 8}{x + 3} - 3\]
			\begin{align*}
				\intertext{$x \neq -3$}
				5 &= x + 8 - 3x - 9 \\
				2x &= -6 \\
				x &= -3 \\
				\Aboxed{x &\in \emptyset}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3x}{x + 2} = \frac{x + 4}{x}\]
			\begin{align*}
				\intertext{$x \notin \{-2, 0\}$}
				3x^2 &= (x + 4)(x + 2) \\
				3x^2 &= x^2 + 2x + 4x + 8 \\
				2x^2 - 6x - 8 &= 0 \\
				x^2 - 3x - 4 &= 0 \\
				(x - 4)(x + 1) &= 0 \\
				\Aboxed{x &\in \{-1, 4\}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} = \frac{3x - 8}{x^2 - 3x - 10}\]
			\begin{align*}
				\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} &= \frac{3x - 8}{(x + 2)(x - 5)} \\
				\intertext{$x \notin \{-2, 5\}$}
				(x + 4)(x - 5) - (2x - 3)(x + 2) &= 3x - 8 \\
				x^2 - x - 20 - 2x^2 - x + 6 &= 3x - 8 \\
				x^2 + 5x + 6 &= 0 \\
				(x + 3)(x + 2) &= 0 \\
				x &\in \{-3, -2\} \\
				\Aboxed{x &= -3}
			\end{align*}
			\hrule{}

		\subsection{}
			\[1 + \frac{5x}{6} = \frac{2x^2}{3}\]
			\begin{align*}
				6 + 5x &= 4x^2 \\
				4x^2 - 5x - 6 &= 0 \\
				4x^2 - 8x + 3x - 6 &= 0 \\
				(4x + 3)(x - 2) &= 0 \\
				\Aboxed{x &\in \left\{-\frac{3}{4}, 2\right\}}
			\end{align*}

		\subsection{}
			\[\frac{x}{2x + 3} = x + 4\]
			\begin{align*}
				\intertext{$x \neq -\frac{3}{2}$}
				x &= (x + 4)(2x + 3) \\
				x &= 2x^2 + 11x + 12 \\
				2x^2 + 10x + 12 &= 0 \\
				x^2 + 5x + 6 &= 0 \\
				(x + 3)(x + 2) &= 0 \\
				\Aboxed{x &\in \{-3, -2\}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\sqrt{x} + 5 = 3\]
			\begin{align*}
				\sqrt{x} &= -2 \\
				\intertext{$\sqrt{x}$ cannot be negative for $x \in \mathbb{R}$.}
				\Aboxed{x &\in \emptyset}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\sqrt{3x + 7} - x = 1\]
			\begin{align*}
				\sqrt{3x + 7} &= x + 1 \\
				3x + 7 &= x^2 + 2x + 1 \\
				x^2 - x - 6 &= 0 \\
				(x - 3)(x + 2) &= 0 \\
				x &\in \{-2, 3\} \\
				\intertext{Check $x = -2$.}
				\sqrt{3(-2) + 7} + 2 &= 1 \\
				3 &\neq 1 \\
				\intertext{Check $x = 3$.}
				\sqrt{3(3) + 7} - 3 &= 1 \\
				1 &= 1 \\
				\Aboxed{x &= 3}
			\end{align*}
			\pagebreak{}

		\subsection{}
			\[\sqrt{\sqrt{5x - 1} - x} = 1\]
			\begin{align*}
				\sqrt{5x - 1} - x &= 1 \\
				\sqrt{5x - 1} &= x + 1 \\
				5x - 1 &= x^2 + 2x + 1 \\
				x^2 - 3x + 2 &= 0 \\
				(x - 2)(x - 1) &= 0 \\
				x &\in \{1, 2\} \\
				\intertext{Check $x = 1$.}
				\sqrt{\sqrt{5(1) - 1} - 1} &= 1 \\
				1 &= 1 \\
				\intertext{Check $x = 2$.}
				\sqrt{\sqrt{5(2) - 1} - 2} &= 1 \\
				1 &= 1 \\
				\Aboxed{x &\in \{1, 2\}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\sqrt{5w + 1} - \sqrt{3w} - 1 = 0\]
			\begin{align*}
				\sqrt{5w + 1} &= \sqrt{3w} + 1 \\
				5w + 1 &= 3w + 2\sqrt{3w} + 1 \\
				2w &= 2\sqrt{3w} \\
				w &= \sqrt{3w} \\
				w^2 - 3w &= 0 \\
				w(w - 3) &= 0 \\
				w &\in \{0, 3\} \\
				\intertext{Check $w = 0$.}
				\sqrt{5(0) + 1} - \sqrt{3(0)} - 1 &= 0 \\
				0 &= 0 \\
				\intertext{Check $w = 3$.}
				\sqrt{5(3) + 1} - \sqrt{3(3)} - 1 &= 0 \\
				0 &= 0 \\
				\Aboxed{w &\in \{0, 3\}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\sqrt{x - 2} = 2\]
			\begin{align*}
				x - 2 &= 4 \\
				x &= 6 \\
				\intertext{Check $x = 6$.}
				\sqrt{6 - 2} &= 2 \\
				2 &= 2 \\
				\Aboxed{x &= 6}
			\end{align*}

		\subsection{}
			\[\sqrt{2x + 5} + x = 5\]
			\begin{align*}
				\sqrt{2x + 5} &= 5 - x \\
				2x + 5 &= 25 - 10x + x^2 \\
				x^2 - 12x + 20 &= 0 \\
				(x - 10)(x - 2) &= 0 \\
				x \in \{2, 10\} \\
				\intertext{Check $x = 2$.}
				\sqrt{2(2) + 5} + 2 &= 5 \\
				5 &= 5 \\
				\intertext{Check $x = 10$.}
				\sqrt{2(10) + 5} + 10 &= 5 \\
				15 &\neq 5 \\
				\Aboxed{x &= 2}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\sqrt{x^2 - 8x} = 3\]
			\begin{align*}
				x^2 - 8x &= 9 \\
				x^2 - 8x - 9 &= 0 \\
				(x - 9)(x + 1) &= 0 \\
				x &\in \{-1, 9\} \\
				\intertext{Check $x = -1$.}
				\sqrt{{(-1)}^2 - 8(-1)} &= 3 \\
				3 &= 3 \\
				\intertext{Check $x = 9$.}
				\sqrt{9^2 - 8(9)} &= 3 \\
				3 &= 3 \\
				\Aboxed{x &\in \{-1, 9\}}
			\end{align*}
			\pagebreak{}

		\subsection{}
			\[\sqrt{x + \sqrt{x - 2}} = 2\]
			\begin{align*}
				x + \sqrt{x - 2} &= 4 \\
				\sqrt{x - 2} &= 4 - x \\
				x - 2 &= 16 - 8x + x^2 \\
				x^2 - 9x + 18 &= 0 \\
				(x - 6)(x - 3) &= 0 \\
				x &\in \{3, 6\} \\
				\intertext{Check $x = 3$.}
				\sqrt{3 + \sqrt{3 - 2}} &= 2 \\
				2 &= 2 \\
				\intertext{Check $x = 6$.}
				\sqrt{6 + \sqrt{6 - 2}} &= 2 \\
				\sqrt{8} &\neq 2 \\
				\Aboxed{x &= 3}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\sqrt{x - 1} - \sqrt{x - 2} - 1 = 0\]
			\begin{align*}
				\sqrt{x - 1} &= \sqrt{x - 2} + 1 \\
				x - 1 &= x - 2 - 2\sqrt{x - 2} + 1 \\
				2\sqrt{x - 2} &= 0 \\
				4x - 8 &= 0 \\
				4x &= 8 \\
				x &= 2 \\
				\intertext{Check $x = 2$.}
				\sqrt{2 - 1} - \sqrt{2 - 2} - 1 &= 0 \\
				0 &= 0 \\
				\Aboxed{x &= 2}
			\end{align*}
			\pagebreak{}

		\subsection{}
			\[2\sqrt{x - 1} - \sqrt{x - 2} - 2 = 0\]
			\begin{align*}
				2\sqrt{x - 1} &= \sqrt{x - 2} + 2 \\
				4x - 4 &= x - 2 + 4\sqrt{x - 2} + 4 \\
				3x - 6 &= 4\sqrt{x - 2} \\
				9x^2 - 36x + 36 &= 16x - 32 \\
				9x^2 - 52x + 68 &= 0 \\
				9x^2 - 18x - 34x + 68 &= 0 \\
				(9x - 34)(x - 2) &= 0 \\
				x &\in \left\{2, \frac{34}{9}\right\} \\
				\intertext{Check $x = 2$.}
				2\sqrt{2 - 1} - \sqrt{2 - 2} - 2 &= 0 \\
				0 &= 0 \\
				\intertext{Check $x = \frac{34}{9}$}
				2\sqrt{\frac{34}{9} - 1} - \sqrt{\frac{34}{9} - 2} - 2 &= 0 \\
				2\sqrt{\frac{34}{9} - \frac{9}{9}} - \sqrt{\frac{34}{9} - \frac{18}{9}} - 2 &= 0 \\
				2\sqrt{\frac{25}{9}} - \sqrt{\frac{16}{9}} - 2 &= 0 \\
				\frac{10}{3} - \frac{4}{3} - 2 &= 0 \\
				0 &= 0 \\
				\Aboxed{x &\in \left\{2, \frac{34}{9}\right\}}
			\end{align*}
			\pagebreak{}

		\subsection{}
			\[\sqrt{3 - 3x} - \sqrt{3x + 2} = 3\]
			\begin{align*}
				\sqrt{3 - 3x} &= \sqrt{3x + 2} + 3 \\
				3 - 3x &= 3x + 2 + 6\sqrt{3x + 2} + 9 \\
				-8 - 6x &= 6\sqrt{3x + 2} \\
				64 + 96x + 36x^2 &= 108x + 72 \\
				36x^2 - 12x - 8 &= 0 \\
				9x^2 - 3x - 2 &= 0 \\
				9x^2 - 6x + 3x - 2 &= 0 \\
				(3x + 1)(3x - 2) &= 0 \\
				x &\in \left\{-\frac{1}{3}, \frac{2}{3}\right\} \\
				\intertext{Check $x = -\frac{1}{3}$.}
				\sqrt{3 - 3\left(-\frac{1}{3}\right)} - \sqrt{3\left(-\frac{1}{3}\right) + 2} &= 3 \\
				\sqrt{4} - \sqrt{1} &= 3 \\
				1 &\neq 3 \\
				\intertext{Check $x = \frac{2}{3}$.}
				\sqrt{3 - 3\left(\frac{2}{3}\right)} - \sqrt{3\left(\frac{2}{3}\right) + 2} &= 3 \\
				\sqrt{1} - \sqrt{4} &= 3 \\
				-1 &\neq 3 \\
				\Aboxed{x &\in \emptyset}
			\end{align*}
			\pagebreak{}

		\subsection{}
			\[2 + 2\sqrt[3]{x - 1} = x\]
			\begin{align*}
				2\sqrt[3]{x - 1} &= x - 2 \\
				8x - 8 &= x^3 - 6x^2 + 12x - 8 \\
				x^3 - 6x^2 + 4x &= 0 \\
				x(x^2 - 6x + 4) &= 0 \\
				\intertext{$x = 0$ may be a solution. Solve for the zeroes of $x^2 - 6x + 4$.}
				x &= \frac{6 \pm \sqrt{{(-6)}^2 - 4(1)(4)}}{2(1)} \\
				x &= \frac{6 \pm \sqrt{36 - 16}}{2} \\
				x &= \frac{6 \pm 2\sqrt{5}}{2} \\
				x &= 3 \pm \sqrt{5} \\
				x &\in \{0, 3 \pm \sqrt{5}\} \\
				\intertext{Check $x = 0$.}
				2 + 2\sqrt[3]{0 - 1} &= 0 \\
				0 &= 0 \\
				\intertext{Check $x = 3 + \sqrt{5}$.}
				2 + 2\sqrt[3]{3 + \sqrt{5} - 1} &= 3 + \sqrt{5} \\
				2\sqrt[3]{2 + \sqrt{5}} &= 1 + \sqrt{5} \\
				16 + 8\sqrt{5} &= 1 + 3\sqrt{5} + 15 + 5\sqrt{5} \\
				16 + 8\sqrt{5} &= 16 + 8\sqrt{5} \\
				\intertext{Check $x = 3 - \sqrt{5}$.}
				2 + 2\sqrt[3]{3 - \sqrt{5} - 1} &= 3 - \sqrt{5} \\
				2\sqrt[3]{2 - \sqrt{5}} &= 1 - \sqrt{5} \\
				16 - 8\sqrt{5} &= 1 - 3\sqrt{5} + 15 - 5\sqrt{5} \\
				16 - 8\sqrt{5} &= 16 - 8\sqrt{5} \\
				\Aboxed{x &\in \{0, 3 \pm \sqrt{5}\}}
			\end{align*}
			\pagebreak{}

		\subsection{}
			\[\sqrt[4]{3x^2 - 3x + 1} = \sqrt{x - 2}\]
			\begin{align*}
				\sqrt{3x^2 - 3x + 1} &= x - 2 \\
				3x^2 - 3x + 1 &= x^2 - 4x + 4 \\
				2x^2 + x - 3 &= 0 \\
				2x^2 - 2x + 3x - 3 &= 0 \\
				(2x + 3)(x - 1) &= 0 \\
				x &\in \left\{-\frac{3}{2}, 1\right\} \\
				\intertext{Check $x = -\frac{3}{2}$.}
				\sqrt[4]{3{\left(-\frac{3}{2}\right)}^2 - 3\left(-\frac{3}{2}\right) + 1} &= \sqrt{-\frac{3}{2} - 2} \\
				\sqrt[4]{3\left(\frac{9}{4}\right) - 3\left(-\frac{6}{4}\right) + \frac{4}{4}} &= \sqrt{-\frac{3}{2} - \frac{4}{2}} \\
				\sqrt[4]{\frac{27}{4} + \frac{18}{4} + \frac{4}{4}} &= i\sqrt{\frac{7}{2}} \\
				\sqrt[4]{\frac{49}{4}} &= i\sqrt{\frac{7}{2}} \\
				\sqrt{\frac{7}{2}} &\neq i\sqrt{\frac{7}{2}} \\
				\intertext{Check $x = 1$.}
				\sqrt[4]{3{(1)}^2 - 3(1) + 1} &= \sqrt{1 - 2} \\
				1 &\neq i \\
				\Aboxed{x &\in \emptyset}
			\end{align*}
