\section{}
Find the solution set of the following inequalities.

\subsection{}
\aproblem{
	\[x - 2 < 4 - x\]
}{
	2x &< 6 \\
	x &< 3
	\afinal{$x \in \ointerval{-\infty}{3}$}
}

\subsection{}
\aproblem{
	\[x + 1 > \dfrac{2x - 4}{5}\]
}{
	5x + 5 &> 2x - 4 \\
	3x &> -9 \\
	x &> -3
	\afinal{$x \in \ointerval{-3}{+\infty}$}
}

\subsection{}
\aproblem{
	\[\dfrac{2x - 3}{4} \geq \dfrac{7x}{2}\]
}{
	2x - 3 &\geq 14x \\
	12x &\leq -3 \\
	x &\leq -\dfrac{3}{12} \\
	x &\leq -\dfrac{1}{4}
	\afinal{$x \in \linterval[scaled]{-\infty}{-\dfrac{1}{4}}$}
}

\subsection{}
\aproblem{
	\[-2 < 5 + 3x < 20\]

\subsection{}
\aproblem{
	\[2x - 1 \leq 3x \leq 4x + 1\]
}{
	\intertext{Solve left inequality.}
	2x - 1 &\leq 3x \\
	x &\geq 1 \\
	x  &\in \rinterval{1}{+\infty}
	\intertext{Solve right inequality.}
	3x &\leq 4x + 1 \\
	x &\geq -1 \\
	x &\in \rinterval{-1}{+\infty}
	\intertext{Intersect intervals.}
	\afinal{$x \in \rinterval{-1}{+\infty}$}
}
