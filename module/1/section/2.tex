\aproblem{
	Find all real values of $k$ such that there is exactly one solution to the equation
	\[x^2 - kx + k = x - 2\]
}{
	x^2 - (k + 1)x + (k + 2) &= 0
	\intertext{\aref*{qdi}, we want the discriminant to be zero for it to have only one solution.}
	{(k + 1)}^2 - 4(1)(k + 2) &= 0 \\
	k^2 + 2k + 1 - 4k - 8 &= 0 \\
	k^2 - 2k - 7 &= 0
	\intertext{\aref*{qqf}}
	k &= \dfrac{2 \pm \sqrt{{(-2)}^2 - 4(1)(-7)}}{2(1)} \\
	k &= \dfrac{2 \pm \sqrt{4 + 28}}{2} \\
	k &= \dfrac{2 \pm 4\sqrt{2}}{2}
	\afinal{$k = 1 \pm 2\sqrt{2}$}
}
