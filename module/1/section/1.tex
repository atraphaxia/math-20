\section{}
Solve the following equations.

\subsection{}
\aproblem{
	\[x + 9 = 5 - 3x\]
}{
	4x &= -4
	\afinal{$x = -1$}
}

\subsection{}
\aproblem{
	\[\dfrac{2x + 3}{4} - \dfrac{x + 1}{2} = -\dfrac{1}{3}\]
}{
	6x + 9 - 6x - 6 &= -4 \\
	3 &= -4
	\afinal{$x \in \emptyset$}
}

\subsection{}
\aproblem{
	\[3x^2 - 2x + 1 = 0\]
}{
	\intertext{\aref*{qqf}}
	x &= \dfrac{2 \pm \sqrt{{(-2)}^2 - 4(3)(1)}}{2(3)} \\
	x &= \dfrac{2 \pm \sqrt{4 - 12}}{6} \\
	x &= \dfrac{2 \pm \sqrt{-8}}{6} \\
	x &= \dfrac{2 \pm 2i\sqrt{2}}{6}
	\afinal{$x = \dfrac{1 \pm i\sqrt{2}}{3}$}
}

\subsection{}
\aproblem{
	\[4x^2 + 2x = 2\]
}{
	2x^2 + x - 1 &= 0 \\
	2x^2 + 2x - x - 1 &= 0 \\
	(2x - 1)(x + 1) &= 0
	\afinal{$x \in \ab\{-1, \dfrac{1}{2}\}$}
}

\subsection{}
\aproblem{
	\[16x^2 + 9 = 24x\]
}{
	16x^2 - 24x + 9 &= 0 \\
	{(4x - 3)}^2 &= 0
	\afinal{$x = \dfrac{3}{4}$}
}

\subsection{}
\aproblem{
	\[\dfrac{x}{x - 1} + \dfrac{x - 5}{x^2 + 2x - 3} = \dfrac{1}{x + 3}\]
}{
	\dfrac{x}{x - 1} + \dfrac{x - 5}{(x - 1)(x + 3)} &= \dfrac{1}{x + 3}
	\intertext{$x \notin \{-3, 1\}$}
	x^2 + 3x + x - 5 &= x - 1 \\
	x^2 + 3x - 4 &= 0 \\
	(x + 4)(x - 1) &= 0 \\
	x &\in \{-4, 1\}
	\afinal{$x = -4$}
}

\subsection{}

	\afinal{$x = 9$}

\subsection{}

\subsection{}

	-5y^2 - y &= -5y^2 + 5
	\afinal{$y = -5$}
}

\subsection{}
