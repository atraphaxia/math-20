\section{}
Perform the indicated operations and simplify.

\subsection{}
\aproblem{
	\[\dfrac{b^\frac{2}{5}b^\frac{1}{3}}{b^\frac{2}{15}}\]
}{
	\intertext{\aref*{em, fas}}
	&= \dfrac{b^\frac{6 + 5}{15}}{b^\frac{2}{15}} \\
	&= \dfrac{b^\frac{11}{15}}{b^\frac{2}{15}}
	\intertext{\aref*{ed, fas}}
	&= b^\frac{11 - 2}{15} \\
	&= b^\frac{9}{15}
	\afinal{$= b^\frac{3}{5}$}
}

\subsection{}
\aproblem{
	\[\dfrac{24c^{-\frac{1}{2}}d^\frac{2}{3}}{18c^{-\frac{1}{7}}d^{-\frac{3}{5}}}\]
}{
	\intertext{\aref*{ed, fas}}
	&= \dfrac{4}{3}c^\frac{-7 + 2}{14}d^\frac{10 + 9}{15} \\
	&= \dfrac{4}{3}c^{-\frac{5}{14}}d^\frac{19}{15} \\
	\intertext{\aref*{ene}}
	&= \dfrac{4d^\frac{19}{15}}{3c^\frac{5}{14}}
}

\subsection{}
\aproblem{
	\[\dfrac{{(2^\frac{2}{3}x^\frac{1}{2}y^\frac{1}{6}z)}^6}{4x^\frac{2}{3}yz^\frac{9}{2}}\]

	\intertext{\aref*{eop, en, fm}}
	&= \dfrac{16x^3yz^6}{4x^\frac{2}{3}yz^\frac{9}{2}}
	\intertext{\aref*{ed, fas}}
	&= 4x^\frac{9 - 2}{3}z^\frac{12 - 9}{2}
	\afinal{$= 4x^\frac{7}{3}z^\frac{3}{2}$}
}

\subsection{}
\aproblem{
	\[{\ab(\dfrac{r^\frac{4}{3}s^\frac{1}{2}t^{-2}}{rs^\frac{2}{7}t^\frac{3}{5}})}^{-10}\]
}{
	\intertext{\aref*{ed, fas}}
	&= {(r^\frac{4 - 3}{3}s^\frac{7 - 4}{14}t^\frac{-10 - 3}{5})}^{-10} \\
	&= {(r^\frac{1}{3}s^\frac{3}{14}t^{-\frac{13}{5}})}^{-10}
	\intertext{\aref*{eop, en}}
	&= r^{-\frac{10}{3}}s^{-\frac{30}{14}}t^{26} \\
	&= r^{-\frac{10}{3}}s^{-\frac{15}{7}}t^{26}
	\intertext{\aref*{ene}}
	\afinal{$= \dfrac{t^{26}}{r^\frac{10}{3}s^\frac{15}{7}}$}
}

\subsection{}
\aproblem{
	\[\sqrt{\dfrac{25c^5}{12a^3b}}\]
}{
	\intertext{\aref*{roq}}
	&= \dfrac{\sqrt{25c^5}}{\sqrt{12a^3b}}
	\intertext{\aref*{rop}}
	&= \dfrac{\sqrt{25c^4}\sqrt{c}}{\sqrt{4a^2}\sqrt{3ab}} \\
	&= \dfrac{5c^2\sqrt{c}}{2a\sqrt{3ab}}
	\intertext{\aref*{rr}}
	&= \dfrac{5c^2\sqrt{c}}{2a\sqrt{3ab}} \cdot \dfrac{\sqrt{3ab}}{\sqrt{3ab}}
	\intertext{\aref*{fm}}
	\afinal{$= \dfrac{5c^2\sqrt{3abc}}{6a^2b}$}
}

\subsection{}
\aproblem{
	\[\dfrac{2x^2y}{\sqrt[3]{12x^2y^4}}\]
}{
	\intertext{\aref*{rop}}
	&= \dfrac{2x^2y}{\sqrt[3]{y^3}\sqrt[3]{12x^2y}} \\
	&= \dfrac{2x^2}{\sqrt[3]{12x^2y}}
	\intertext{\aref*{rr}}
	&= \dfrac{2x^2}{\sqrt[3]{12x^2y}} \cdot \dfrac{\sqrt[3]{144xy^2}}{\sqrt[3]{12^2xy^2}}
	\intertext{\aref*{fm}}
	&= \dfrac{2x^2\sqrt[3]{144xy^2}}{12xy}
	\intertext{\aref*{rop}}
	&= \dfrac{2x^2\sqrt[3]{8}\sqrt[3]{18xy^2}}{12xy} \\
	&= \dfrac{4x^2\sqrt[3]{18xy^2}}{12xy}
	\afinal{$= \dfrac{x\sqrt[3]{18xy^2}}{3y}$}
}

\subsection{}
\aproblem{
	\[\dfrac{x^2 - 1}{\sqrt{x + 1}}\]
}{
	\intertext{\aref*{fdts}}
	&= \dfrac{(x - 1)(x + 1)}{\sqrt{x + 1}}
	\intertext{\aref*{rr}}
	&= \dfrac{(x - 1)(x + 1)}{\sqrt{x + 1}} \cdot \dfrac{\sqrt{x + 1}}{\sqrt{x + 1}}
	\intertext{\aref*{fm}}
	&= \dfrac{(x - 1)(x + 1)\sqrt{x + 1}}{x + 1}
	\afinal{$= (x - 1)\sqrt{x + 1}$}
}

\subsection{}
\aproblem{
	\[\dfrac{x^2 - 2x + 1}{\sqrt{x} + 1}\]
}{
	\intertext{\aref*{fpst}}
	&= \dfrac{{(x - 1)}^2}{\sqrt{x} + 1}
	\intertext{\aref*{rr, fdts}}
	&= \dfrac{{(x - 1)}^2}{\sqrt{x} + 1} \cdot \dfrac{\sqrt{x} - 1}{\sqrt{x} - 1}
	\intertext{\aref*{fm}}
	&= \dfrac{{(x - 1)}^2(\sqrt{x} - 1)}{x - 1}
	\afinal{$= (x - 1)(\sqrt{x} - 1)$}
}

\subsection{}
\aproblem{
	\[\dfrac{6}{\sqrt{3 - \sqrt{3}}}\]
}{
	\intertext{\aref*{rr}}
	&= \dfrac{6}{\sqrt{3 - \sqrt{3}}} \cdot \dfrac{\sqrt{3 - \sqrt{3}}}{\sqrt{3 - \sqrt{3}}}
	\intertext{\aref*{fm}}
	&= \dfrac{6\sqrt{3 - \sqrt{3}}}{3 - \sqrt{3}}
	\intertext{\aref*{rr, fdts}}
	&= \dfrac{6\sqrt{3 - \sqrt{3}}}{3 - \sqrt{3}} \cdot \dfrac{3 + \sqrt{3}}{3 + \sqrt{3}}
	\intertext{\aref*{fm}}
	&= \dfrac{6(3 + \sqrt{3})\sqrt{3 - \sqrt{3}}}{9 - 3}
	\afinal{$= (3 + \sqrt{3})\sqrt{3 - \sqrt{3}}$}
}

	\[\dfrac{1}{\sqrt[3]{4} + \sqrt[3]{-27}}\]
}{

	\intertext{\aref*{fm}}
	&= \dfrac{\sqrt[3]{16} + 3\sqrt[3]{4} + 9}{4 - 27}
	\afinal{$= -\dfrac{\sqrt[3]{16} + 3\sqrt[3]{4} + 9}{23}$}
}

\subsection{}
\aproblem{
	\[\dfrac{2\sqrt{5}}{\sqrt{8}} + \dfrac{9}{\sqrt[3]{16}}\]
}{
	\intertext{\aref*{roq}}
	&= \dfrac{2\sqrt{5}}{\sqrt{4}\sqrt{2}} + \dfrac{9}{\sqrt[3]{8}\sqrt[3]{2}} \\
	&= \dfrac{2\sqrt{5}}{2\sqrt{2}} + \dfrac{9}{2\sqrt[3]{2}}
	\intertext{\aref*{rr}}
	&= \dfrac{2\sqrt{5}}{2\sqrt{2}} \cdot \dfrac{\sqrt{2}}{\sqrt{2}} + \dfrac{9}{2\sqrt[3]{2}} \cdot \dfrac{\sqrt[3]{4}}{\sqrt[3]{2^2}}
	\intertext{\aref*{fm}}
	&= \dfrac{2\sqrt{10}}{4} + \dfrac{9\sqrt[3]{4}}{4}
	\afinal{$= \dfrac{2\sqrt{10} + 9\sqrt[3]{4}}{4}$}
}
