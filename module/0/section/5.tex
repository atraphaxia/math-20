\section{}
Perform the indicated operations and simplify.

\subsection{}
\aproblem{
	\[\overline{1 - i}\]
}{
	\intertext{\aref*{cnc}}
	\afinal{$= 1 + i$}
}

\subsection{}
\aproblem{
	\[\overline{5i + 2}\]
}{
	\intertext{\aref*{cnc}}
	\afinal{$= 2 - 5i$}
}

\subsection{}
\aproblem{
	\[(\overline{5 - i}) - (\overline{5 + i})\]
}{
	\intertext{\aref*{cnc}}
	&= 5 + i - 5 + i
	\afinal{$= 2i$}
}

\subsection{}
\aproblem{
	\[\sqrt{-1} - \sqrt{3} + \sqrt{2} - \sqrt{-4}\]
}{
	\intertext{\aref*{ind}}
	&= \sqrt{2} - \sqrt{3} + i - 2i
	\afinal{$= \sqrt{2} - \sqrt{3} - i$}
}

\subsection{}
\aproblem{
	\[4 - 2\sqrt{-45} - (\overline{3 - \sqrt{-20}})\]
}{
	\intertext{\aref*{cnc}}
	&= 4 - 2\sqrt{-45} - 3 - \sqrt{-20}
	\intertext{\aref*{rop}}
	&= 1 - 2\sqrt{-9}\sqrt{5} - \sqrt{-4}\sqrt{5}
	\intertext{\aref*{ind}}
	&= 1 - 6i\sqrt{5} - 2i\sqrt{5}
	\afinal{$= 1 - 8i\sqrt{5}$}
}

\subsection{}
\aproblem{
	\[\sqrt{-12}\sqrt{-75}\]
}{
	\intertext{\aref*{rop}}
	&= \sqrt{-4}\sqrt{3}\sqrt{-25}\sqrt{3}
	\intertext{\aref*{ind}}
	\afinal{$= -30$}
}

\subsection{}
\aproblem{
	\[3i(i^2 - i^3 + 5i^5 - i^{-2})\]
}{
	\intertext{\aref*{inc}}
	&= 3i(-1 + i + 5i + 1) \\
	&= -3i + 3i^2 + 15i^2 + 3i
	\intertext{\aref*{ind}}
	\afinal{$= -18$}
}

\subsection{}
\aproblem{
	\[(3 - 5i)(7 + 4i)\]
}{
	\intertext{\aref*{cnm}}
	&= (21 + 20) + (12 - 35)i
	\afinal{$= 41 - 23i$}
}

\subsection{}
\aproblem{
	\[(1 + 2i)(3 - 4i)\]
}{
	\intertext{\aref*{cnm}}
	&= (3 + 8) + (-4 + 6)i
	\afinal{$= 11 + 2i$}
}

\subsection{}
\aproblem{
	\[(i - 1)(\overline{5 + 3i})\]
}{
	\intertext{\aref*{cnc}}
	&= (-1 + i)(5 - 3i)
	\intertext{\aref*{cnm}}
	&= (-5 + 3) + (3 + 5)i
	\afinal{$-2 + 8i$}
}

\subsection{}
\aproblem{
	\[{(2i - 3)}^2\]
}{
	&= {(-3 + 2i)}^2
	\intertext{\aref*{cnm}}
	&= (9 - 4) + (-6 - 6)i
	\afinal{$= 5 - 12i$}
}

\subsection{}
\aproblem{
	\[\dfrac{3 + 4i}{2 - i}\]
}{
	\intertext{\aref*{cndv}}
	&= \dfrac{6 - 4}{4 + 1} + \dfrac{8 + 3}{4 + 1}i
	\afinal{$= \dfrac{2}{5} + \dfrac{11}{5}i$}
}

\subsection{}
\aproblem{
	\[\dfrac{\sqrt{-9} + \sqrt{8}}{\sqrt{-1}}\]
}{
	\intertext{\aref*{ind, rop}}
	&= \dfrac{\sqrt{4}\sqrt{2} + 3i}{i} \\
	&= \dfrac{2\sqrt{2} + 3i}{i}
	\intertext{\aref*{ene}}
	&= 3 + 2i^{-1}\sqrt{2}
	\afinal{$= 3 - 2i\sqrt{2}$}
}

\subsection{}
\aproblem{
	\[\dfrac{3i - 2}{3i + 2}\]
}{
	&= \dfrac{-2 + 3i}{2 + 3i}
	\intertext{\aref*{cndv}}
	&= \dfrac{-4 + 9}{4 + 9} + \dfrac{6 + 6}{4 + 9}i
	\afinal{$= \dfrac{5}{13} + \dfrac{12}{13}i$}
}

\subsection{}
\aproblem{
	\[\dfrac{(1 - i)(3i + 1)}{2i - 1}\]
}{
	&= \dfrac{(1 - i)(1 + 3i)}{-1 + 2i}
	\intertext{\aref*{cnm}}
	&= \dfrac{(1 + 3) + (3 - 1)i}{-1 + 2i} \\
	&= \dfrac{4 + 2i}{-1 + 2i}
	\intertext{\aref*{cndv}}
	&= \dfrac{-4 + 4}{1 + 4} + \dfrac{-2 - 8}{1 + 4}i \\
	&= -\dfrac{10}{5}i
	\afinal{$= -2i$}
}

\subsection{}
\aproblem{
	\[\ab(-\dfrac{1}{2} + \dfrac{\sqrt{3}}{2}i)\ab(\dfrac{\sqrt{2}}{2} - \dfrac{\sqrt{2}}{2}i)\]
}{
	\intertext{\aref*{cnm, rop}}
	&= \ab(-\dfrac{\sqrt{2}}{4} + \dfrac{\sqrt{6}}{4}) + \ab(\dfrac{\sqrt{2}}{4} + \dfrac{\sqrt{6}}{4})i
	\afinal{$= \dfrac{\sqrt{6} - \sqrt{2}}{4} + \dfrac{\sqrt{6} + \sqrt{2}}{4}i$}
}

\subsection{}
\aproblem{
	\[\dfrac{7 + i - 4(3 - i)}{6 - 5i^3}\]
}{
	\intertext{\aref*{inc}}
	&= \dfrac{7 + i - 12 + 4i}{6 + 5i} \\
	&= \dfrac{-5 + 5i}{6 + 5i}
	\intertext{\aref*{cndv}}
	&= \dfrac{-30 + 25}{36 + 25} + \dfrac{30 + 25}{36 + 25}i
	\afinal{$= -\dfrac{5}{61} + \dfrac{55}{61}i$}
}

\subsection{}
\aproblem{
	\[\dfrac{2 - 2(\overline{i + 1})}{2 - \sqrt{-4}}\]
}{
	\intertext{\aref*{cnc, ind}}
	&= \dfrac{2 - 2(1 - i)}{2 - 2i} \\
	&= \dfrac{1 - 1 + i}{1 - i} \\
	&= \dfrac{i}{1 - i}
	\intertext{\aref*{cndv}}
	&= \dfrac{-1}{1 + 1} + \dfrac{1}{1 + 1}i
	\afinal{$= -\dfrac{1}{2} + \dfrac{1}{2}i$}
}
