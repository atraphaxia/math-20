\section{}
Perform the indicated operations and simplify.

\subsection{}
\aproblem{
	\[\dfrac{3a - 3b}{c^2 + 3cd} \cdot \dfrac{2cd + 6d^2}{ac - bc}\]
}{
	\intertext{\aref*{fcmf}}
	&= \dfrac{3(a - b)}{c(c + 3d)} \cdot \dfrac{2d(c + 3d)}{c(a - b)}
	\intertext{\aref*{fm}}
	\afinal{$= \dfrac{6d}{c^2}$}
}

\subsection{}
\aproblem{
	\[\dfrac{x^3 - 8}{x^2 - 6x + 8} \div \dfrac{3x^3 + 6x^2 + 12x}{x^2 - 8x + 16}\]
}{
	\intertext{\aref*{fd, fsdtc, fst, fcmf, fpst}}
	&= \dfrac{(x - 2)(x^2 + 2x + 4)}{(x - 4)(x - 2)} \cdot \dfrac{{(x - 4)}^2}{3x(x^2 + 2x + 4)}
	\intertext{\aref*{fm}}
	\afinal{$= \dfrac{x - 4}{3x}$}
}

\subsection{}
\aproblem{
	\[\dfrac{x + 1}{x} - \dfrac{y + 1}{y}\]
}{
	\intertext{\aref*{fas}}
	&= \dfrac{xy + y - xy - x}{xy}
	\afinal{$= \dfrac{y - x}{xy}$}
}

\subsection{}
\aproblem{
	\[\dfrac{a}{bc} + \dfrac{b}{ac} - \dfrac{c}{ab}\]
}{
	\intertext{\aref*{fas}}
	\afinal{$= \dfrac{a^2 + b^2 - c^2}{abc}$}
}

\subsection{}
\aproblem{
	\[\dfrac{1}{x} - \dfrac{2}{x + y}\]
}{
	\intertext{\aref*{fas}}
	&= \dfrac{x + y - 2x}{x(x + y)}
	\afinal{$= \dfrac{y - x}{x(x + y)}$}
}

\subsection{}

	\[\dfrac{2}{x - y} + \dfrac{3}{y - x}\]
}{
	&= \dfrac{2}{x - y} - \dfrac{3}{x - y}
	\afinal{$= -\dfrac{1}{x - y}$}
}

\subsection{}
\aproblem{
	\[\dfrac{2}{(a + b)(a - b)} + \dfrac{1}{{(a - b)}^2}\]
}{
	\intertext{\aref*{fas}}
	&= \dfrac{2a - 2b + a + b}{(a + b){(a - b)}^2}
	\afinal{$= \dfrac{3a - b}{(a + b){(a - b)}^2}$}
}

\subsection{}
\aproblem{
	\[\dfrac{x^2}{x(x - 1)} + \dfrac{1}{x} - \dfrac{1}{x - 1}\]
}{
	\intertext{\aref*{fas}}
	&= \dfrac{x^2 + x - 1 - x}{x(x - 1)} \\
	&= \dfrac{x^2 - 1}{x(x - 1)}
	\intertext{\aref*{fdts}}
	&= \dfrac{(x - 1)(x + 1)}{x(x - 1)}
	\afinal{$= \dfrac{x + 1}{x}$}
}

\subsection{}
\aproblem{
	\[\dfrac{4}{4x^2 - 9} - \dfrac{3}{2x^2 - x - 3}\]
}{
	\intertext{\aref*{fdts, fct}}
	&= \dfrac{4}{(2x - 3)(2x + 3)} - \dfrac{3}{2x^2 - 3x + 2x - 3} \\
	&= \dfrac{4}{(2x - 3)(2x + 3)} - \dfrac{3}{(x + 1)(2x - 3)}
	\intertext{\aref*{fas}}
	&= \dfrac{4x + 4 - 6x - 9}{(2x - 3)(2x + 3)(x + 1)}
	\afinal{$= -\dfrac{2x + 5}{(2x - 3)(2x + 3)(x + 1)}$}
}

\subsection{}
\aproblem{
	\[\dfrac{3}{2x - 6} - \dfrac{9}{x^2 - 9}\]
}{
	\intertext{\aref*{fcmf, fdts}}
	&= \dfrac{3}{2(x - 3)} - \dfrac{9}{(x - 3)(x + 3)}
	\intertext{\aref*{fas}}
	&= \dfrac{3x + 9 - 18}{2(x - 3)(x + 3)} \\
	&= \dfrac{3x - 9}{2(x - 3)(x + 3)}
	\intertext{\aref*{fcmf}}
	&= \dfrac{3(x - 3)}{2(x - 3)(x + 3)}
	\afinal{$= \dfrac{3}{2(x + 3)}$}
}

\subsection{}
\aproblem{
	\[\dfrac{x - \dfrac{9}{x}}{1 - \dfrac{3}{x}}\]
}{
	\intertext{\aref*{fcf}}
	&= \dfrac{x^2 - 9}{x - 3}
	\intertext{\aref*{fdts}}
	&= \dfrac{(x - 3)(x + 3)}{x - 3}
	\afinal{$= x + 3$}
}

\subsection{}
\aproblem{
	\[\dfrac{2 - \dfrac{7}{x} - \dfrac{4}{x^2}}{3 - \dfrac{13}{x} + \dfrac{4}{x^2}}\]
}{
	\intertext{\aref*{fcf}}
	&= \dfrac{2x^2 - 7x - 4}{3x^2 - 13x + 4}
	\intertext{\aref*{fct}}
	&= \dfrac{2x^2 - 8x + x - 4}{3x^2 - 12x - x + 4} \\
	&= \dfrac{(2x + 1)(x - 4)}{(3x - 1)(x - 4)}
	\afinal{$= \dfrac{2x + 1}{3x - 1}$}
}

\subsection{}
\aproblem{
	\[\dfrac{w - 2 + \dfrac{w - 2}{w + 2}}{w - \dfrac{3w + 12}{w + 2}}\]
}{
	\intertext{\aref*{fcf}}
	&= \dfrac{w^2 + 2w - 2w - 4 + w - 2}{w^2 + 2w - 3w - 12} \\
	&= \dfrac{w^2 + w - 6}{w^2 - w - 12}
	\intertext{\aref*{fst}}
	&= \dfrac{(w + 3)(w - 2)}{(w - 4)(w + 3)}
	\afinal{$= \dfrac{w - 2}{w - 4}$}
}

\subsection{}
\aproblem{
	\[\dfrac{a^2 - b^2}{a^3 - b^3}\]
}{
	\intertext{\aref*{fdts, fsdtc}}
	&= \dfrac{(a - b)(a + b)}{(a - b)(a^2 + ab + b^2)}
	\afinal{$= \dfrac{a + b}{a^2 + ab + b^2}$}
}

\subsection{}
\aproblem{
	\[\dfrac{x^3 - x^2y + xy^2 - y^3}{x^6 + y^6}\]
}{
	\intertext{\aref*{fct, fsdtc}}
	&= \dfrac{(x^2 + y^2)(x - y)}{(x^2 + y^2)(x^4 - x^2y^2 + y^4)}
	\afinal{$= \dfrac{x - y}{x^4 - x^2y^2 + y^4}$}
}

\subsection{}
\aproblem{
	\[\ab(\dfrac{x}{x^2 - 1} - \dfrac{3}{x + 1}) \div \dfrac{2x^2 - x - 3}{x^3 - 1}\]
}{
	\intertext{\aref*{fdts, fd, fct, fsdtc}}
	&= \ab(\dfrac{x}{(x - 1)(x + 1)} - \dfrac{3}{x + 1}) \cdot \dfrac{(x - 1)(x^2 + x + 1)}{2x^2 - 3x + 2x - 3} \\
	&= \ab(\dfrac{x}{(x - 1)(x + 1)} - \dfrac{3}{x + 1}) \cdot \dfrac{(x - 1)(x^2 + x + 1)}{(x + 1)(2x - 3)}
	\intertext{\aref*{fas}}
	&= \dfrac{x - 3x + 3}{(x - 1)(x + 1)} \cdot \dfrac{(x - 1)(x^2 + x + 1)}{(x + 1)(2x - 3)} \\
	&= -\dfrac{2x - 3}{(x - 1)(x + 1)} \cdot \dfrac{(x - 1)(x^2 + x + 1)}{(x + 1)(2x - 3)}
	\intertext{\aref*{fm}}
	\afinal{$= -\dfrac{x^2 + x + 1}{{(x + 1)}^2}$}
}

\subsection{}
\aproblem{
	\[\dfrac{\ab(\dfrac{x}{x + y} + \dfrac{y}{x - y})\ab(\dfrac{x^2 - xy}{x^4 - y^4})}{\dfrac{x}{x^2 + 2xy + y^2}}\]
}{
	\intertext{\aref*{fas, fcmf, fdts, fd, fpst}}
	&= \dfrac{x^2 - xy + xy + y^2}{(x + y)(x - y)} \cdot \dfrac{x(x - y)}{(x^2 - y^2)(x^2 + y^2)} \cdot \dfrac{{(x + y)}^2}{x} \\
	&= \dfrac{x^2 + y^2}{(x + y)(x - y)} \cdot \dfrac{x(x - y)}{(x^2 - y^2)(x^2 + y^2)} \cdot \dfrac{{(x + y)}^2}{x}
	\intertext{\aref*{fdts}}
	&= \dfrac{x^2 + y^2}{(x + y)(x - y)} \cdot \dfrac{x(x - y)}{(x - y)(x + y)(x^2 + y^2)} \cdot \dfrac{{(x + y)}^2}{x}
	\intertext{\aref*{fm}}
	\afinal{$= \dfrac{1}{x - y}$}
}

\subsection{}
\aproblem{
	\[\dfrac{\dfrac{3}{p + q} + \dfrac{1}{p - 2q}}{1 + \dfrac{p - q}{p - 2q}}\]
}{
	\intertext{\aref*{fas, fd}}
	&= \dfrac{3p - 6q + p + q}{(p + q)(p - 2q)} \cdot \dfrac{p - 2q}{p - 2q + p - q} \\
	&= \dfrac{4p - 5q}{(p + q)(p - 2q)} \cdot \dfrac{p - 2q}{2p - 3q}
	\intertext{\aref*{fm}}
	\afinal{$= \dfrac{4p - 5q}{(p + q)(2p - 3q)}$}
}
