\section{}
	Perform the indicated operations and simplify.

	\subsection{}
		\[\overline{1 - i}\]
		\begin{align*}
			\Aboxed{{} &= 1 + i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\overline{5i - 2}\]
		\begin{align*}
			\Aboxed{{} &= 2 - 5i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\overline{5 - i} - \overline{5 + i}\]
		\begin{align*}
			&= 5 + i - (5 - i) \\
			\Aboxed{{} &= 2i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\sqrt{-1} - \sqrt{3} + \sqrt{2} - \sqrt{-4}\]
		\begin{align*}
			&= \sqrt{2} - \sqrt{3} + i - 2i \\
			\Aboxed{{} &= \sqrt{2} - \sqrt{3} - i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[(4 - 2\sqrt{-45}) - \overline{(3 - \sqrt{-20})}\]
		\begin{align*}
			&= 4 - 6i\sqrt{5} - \overline{(3 - 2i\sqrt{5})} \\
			&= 4 - 6i\sqrt{5} - (3 + 2i\sqrt{5}) \\
			\Aboxed{{} &= 1 - 8i\sqrt{5}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\sqrt{-12}\sqrt{-75}\]
		\begin{align*}
			&= 2i\sqrt{3} \cdot 5i\sqrt{3} \\
			\Aboxed{{} &= -30}
		\end{align*}
		\hrule{}

	\subsection{}
		\[3i(i^2 - i^3 + 5i^5 - i^{-2})\]
		\begin{align*}
			&= 3i(-1 + i + 5i + 1) \\
			&= 3i(6i) \\
			\Aboxed{{} &= -18}
		\end{align*}

	\subsection{}
		\[(3 - 5i)(7 + 4i)\]
		\begin{align*}
			&= 21 - 35i + 12i + 20 \\
			\Aboxed{{} &= 41 - 23i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[(1 + 2i)(3 - 4i)\]
		\begin{align*}
			&= 3 - 4i + 6i + 8 \\
			\Aboxed{{} &= 11 + 2i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[(i - 1)\overline{(5 + 3i)}\]
		\begin{align*}
			&= (i - 1)(5 - 3i) \\
			&= 5i + 3 - 5 + 3i \\
			\Aboxed{{} &= -2 + 8i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[{(2i - 3)}^2\]
		\begin{align*}
			&= 4i^2 - 12i + 9 \\
			\Aboxed{{} &= 5 - 12i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{3 + 4i}{2 - i}\]
		\begin{align*}
			&= \frac{(3 + 4i)(2 + i)}{5} \\
			&= \frac{6 + 8i + 3i - 4}{5} \\
			\Aboxed{{} &= \frac{2}{5} + \frac{11}{5}i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{\sqrt{-9} + \sqrt{8}}{\sqrt{-1}}\]
		\begin{align*}
			&= \frac{3i + 2\sqrt{2}}{i} \\
			&= 3 + 2i^{-1}\sqrt{2} \\
			\Aboxed{{} &= 3 - 2i\sqrt{2}}
		\end{align*}

	\subsection{}
		\[\frac{3i - 2}{3i + 2}\]
		\begin{align*}
			&= \frac{(3i - 2)(2 - 3i)}{13} \\
			&= \frac{6i + 9 - 4 + 6i}{13} \\
			\Aboxed{{} &= \frac{5}{13} + \frac{12}{13}i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{(1 - i)(3i + 1)}{2i - 1}\]
		\begin{align*}
			&= \frac{(3i + 1 + 3 - i)(-1 - 2i)}{5} \\
			&= \frac{(4 + 2i)(-1 - 2i)}{5} \\
			&= \frac{-4 + -8i - 2i + 4}{5} \\
			\Aboxed{{} &= -10i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\left(\frac{-1}{2} + \frac{\sqrt{3}}{2}i\right)\left(\frac{\sqrt{2}}{2} - \frac{\sqrt{2}}{2}i\right)\]
		\begin{align*}
			&= -\frac{\sqrt{2}}{4} + \frac{\sqrt{2}}{4}i + \frac{\sqrt{6}}{4}i + \frac{\sqrt{6}}{4} \\
			\Aboxed{{} &= \frac{\sqrt{6} - \sqrt{2}}{4} + \frac{\sqrt{6} + \sqrt{2}}{4}i}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{7 + i - 4(3 - i)}{6 - 5i^3}\]
		\begin{align*}
			&= \frac{7 + i - 12 + 4i}{6 + 5i} \\
			&= \frac{(-5 + 5i)(6 - 5i)}{61} \\
			&= \frac{-30 + 25i + 30i + 25}{61} \\
			\Aboxed{{} &= -\frac{5}{61} + \frac{55}{61}i}
		\end{align*}
		\pagebreak{}

	\subsection{}
		\[\frac{2 - 2\overline{(i + 1)}}{2 - \sqrt{-4}}\]
		\begin{align*}
			&= \frac{2 - 2(1 - i)}{2 - 2i} \\
			&= \frac{2i(2 + 2i)}{8} \\
			&= \frac{i(1 + i)}{2} \\
			\Aboxed{{} &= -\frac{1}{2} + \frac{1}{2}i}
		\end{align*}
