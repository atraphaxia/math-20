\section{}
	Perform the indicated operations and simplify.

	\subsection{}
		\[\frac{b^{\frac{2}{5}} \cdot b^{\frac{1}{3}}}{b^{\frac{2}{15}}}\]
		\begin{align*}
			&= b^{\frac{6}{15}} \cdot b^{\frac{5}{15}} \div b^{\frac{2}{15}} \\
			\Aboxed{{} &= b^{\frac{3}{5}}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{24c^{-\frac{1}{2}}d^{\frac{2}{3}}}{18c^{-\frac{1}{7}}d^{-\frac{3}{5}}}\]
		\begin{align*}
			&= \frac{4c^{-\frac{7}{14}}d^{\frac{10}{15}}}{3c^{-\frac{2}{14}}d^{-\frac{9}{15}}} \\
			\Aboxed{{} &= \frac{4d^{\frac{19}{15}}}{3c^{\frac{5}{14}}}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{{(2^{\frac{2}{3}}x^{\frac{1}{2}}y^{\frac{1}{6}}z)}^6}{4x^{\frac{2}{3}}yz^{\frac{9}{2}}}\]
		\begin{align*}
			&= \frac{2^4x^3yz^6}{4x^{\frac{2}{3}}yz^{\frac{9}{2}}} \\
			&= \frac{16x^{\frac{9}{3}}z^{\frac{12}{2}}}{4x^{\frac{2}{3}}z^{\frac{9}{2}}} \\
			\Aboxed{{} &= 4x^{\frac{7}{3}}z^{\frac{3}{2}}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[(u^{\frac{1}{3}} + {(uv)}^{\frac{1}{6}} + v^{\frac{1}{3}})(u^{\frac{1}{6}} - v^{\frac{1}{6}})\]
		\begin{align*}
			\Aboxed{{} &= u^{\frac{1}{2}} - v^{\frac{1}{2}}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[{\left(\frac{r^{\frac{4}{3}}s^{\frac{1}{2}}t^{-2}}{rs^{\frac{2}{7}}t^{\frac{3}{5}}}\right)}^{-10}\]
		\begin{align*}
			&= \frac{r^{-\frac{40}{3}}s^{-5}t^{20}}{r^{-10}s^{-\frac{20}{7}}t^{-6}} \\
			&= \frac{r^{-\frac{40}{3}}s^{-\frac{35}{7}}t^{20}}{r^{-\frac{30}{3}}s^{-\frac{20}{7}}t^{-6}} \\
			\Aboxed{{} &= \frac{t^{26}}{r^{\frac{10}{3}}s^{\frac{15}{7}}}}
		\end{align*}

	\subsection{}
		\[\sqrt[3]{-8^4}\]
		\begin{align*}
			&= \sqrt[3]{-8^3}\sqrt[3]{8} \\
			\Aboxed{{} &= -16}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\sqrt{384x^3y^6}\]
		\begin{align*}
			&= xy^3\sqrt{64}\sqrt{6x} \\
			\Aboxed{{} &= 8xy^3\sqrt{6x}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\sqrt[4]{9x^8}\]
		\begin{align*}
			&= x^2\sqrt[4]{3^2} \\
			\Aboxed{{} &= x^2\sqrt{3}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\sqrt{\sqrt[3]{4a^4b^4}}\]
		\begin{align*}
			&= \sqrt[6]{2^2a^4b^4} \\
			\Aboxed{{} &= \sqrt[3]{2a^2b^2}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\sqrt{\frac{25c^5}{12a^3b}}\]
		\begin{align*}
			&= \frac{5c^2\sqrt{c}}{2a\sqrt{3ab}} \\
			\Aboxed{{} &= \frac{5c^2\sqrt{3abc}}{6a^2b}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{2x^2y}{\sqrt[3]{12x^2y^4}}\]
		\begin{align*}
			&= \frac{2x^2}{\sqrt[3]{12x^2y}} \\
			&= \frac{2x^2\sqrt[3]{{(12x^2y)}^2}}{12x^2y} \\
			&= \frac{\sqrt[3]{144x^4y^2}}{6y} \\
			&= \frac{2x\sqrt[3]{18xy^2}}{6y} \\
			\Aboxed{{} &= \frac{x\sqrt[3]{18xy^2}}{3y}}
		\end{align*}

	\subsection{}
		\[\sqrt{2} \cdot \sqrt[3]{2}\]
		\begin{align*}
			&= 2^{\frac{1}{2}} \cdot 2^{\frac{1}{3}} \\
			\Aboxed{{} &= 2^{\frac{5}{6}}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{x^2 - 1}{\sqrt{x + 1}}\]
		\begin{align*}
			&= \frac{(x + 1)(x - 1)}{\sqrt{x + 1}} \\
			\Aboxed{{} &= (x - 1)\sqrt{x + 1}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{\sqrt{a^2 - b^2}}{\sqrt{a^2 + b^2}}\]
		\begin{align*}
			&= \frac{\sqrt{a^2 - b^2}\sqrt{a^2 + b^2}}{a^2 + b^2} \\
			\Aboxed{{} &= \frac{\sqrt{a^4 - b^4}}{a^2 + b^2}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\sqrt[4]{36} - \sqrt{54} + \sqrt{96}\]
		\begin{align*}
			&= \sqrt[4]{6^2} - 3\sqrt{6} + 4\sqrt{6} \\
			\Aboxed{{} &= 2\sqrt{6}} \\
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{2 - \sqrt{3}}{3 - \sqrt{2}}\]
		\begin{align*}
			&= \frac{(2 - \sqrt{3})(3 + \sqrt{2})}{7} \\
			\Aboxed{{} &= \frac{6 + 2\sqrt{2} - 3\sqrt{3} - \sqrt{6}}{7}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{\sqrt{6} - \sqrt{5}}{\sqrt{5} + \sqrt{6}}\]
		\begin{align*}
			&= -(\sqrt{6} - \sqrt{5})(\sqrt{5} - \sqrt{6}) \\
			&= 5 - 2\sqrt{5}{\sqrt{6}} + 6 \\
			\Aboxed{{} &= 11 - 2\sqrt{30}}
		\end{align*}

	\subsection{}
		\[\frac{x^2 - 2x + 1}{\sqrt{x} + 1}\]
		\begin{align*}
			&= \frac{{(x - 1)}^2(\sqrt{x} + 1)}{x - 1} \\
			\Aboxed{{} &= (x - 1)(\sqrt{x} + 1)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{6}{\sqrt{3 - \sqrt{3}}}\]
		\begin{align*}
			&= \frac{6\sqrt{3 - \sqrt{3}}}{3 - \sqrt{3}} \\
			&= \frac{6(3 + \sqrt{3})\sqrt{3 - \sqrt{3}}}{6} \\
			\Aboxed{{} &= (3 + \sqrt{3})\sqrt{3 - \sqrt{3}}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{1}{\sqrt[3]{4} + \sqrt[3]{-27}}\]
		\begin{align*}
			&= \frac{1}{\sqrt[3]{4} - 3} \\
			\Aboxed{{} &= -\frac{\sqrt{16} + 3\sqrt[3]{4} + 9}{23}}
		\end{align*}
		\hrule{}

	\subsection{}
		\[\frac{2\sqrt{5}}{\sqrt{8}} + \frac{9}{\sqrt[3]{16}}\]
		\begin{align*}
			&= \frac{2\sqrt{5}}{2\sqrt{2}} + \frac{9}{2\sqrt[3]{2}} \\
			&= \frac{\sqrt{10}}{2} + \frac{9\sqrt[3]{4}}{4} \\
			\Aboxed{{} &= \frac{2\sqrt{10} + 9\sqrt[3]{4}}{4}}
		\end{align*}
