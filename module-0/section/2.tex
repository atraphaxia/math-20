	\subsection{}
		\[4x^3y^2 + 6x^2y^3\]
		\begin{align*}
			\Aboxed{{} &= 2x^2y^2(2x + 3y)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[a^2b^3c^4 - a^3b^4c^5 + 2a^2b^4c^4\]
		\begin{align*}
			\Aboxed{{} &= a^2b^3c^4(1 - abc + 2b)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[9a^2 - 1\]

			\Aboxed{{} &= (3a - 1)(3a + 1)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[16x^2 - 25y^2\]

	\subsection{}
		\[25x^4y^8 - 64z^2\]

	\subsection{}
		\[16x^4 - 1\]
		\begin{align*}

		\end{align*}
		\hrule{}

	\subsection{}
		\[{(s - 2t)}^2 - 4\]

	\subsection{}
		\[{(x + 3)}^2 - {(y + 2)}^2\]
		\begin{align*}
			&= (x + 3 - y - 2)(x + 3 + y + 2) \\
			\Aboxed{{} &= {x - y - 1}(x + y + 5)}
		\end{align*}

	\subsection{}
		\[x^2(x^2 - 1) - 9(x^2 - 1)\]

		\end{align*}
		\hrule{}

	\subsection{}

		\end{align*}
		\hrule{}

	\subsection{}

	\subsection{}

	\subsection{}
		\[7s^6 - 9s^3 + 2\]
		\begin{align*}
			&= 7s^6 - 7s^3 - 2s^3 + 2 \\
			&= (7s^3 - 2)(s^3 - 1) \\
			\Aboxed{{} &= (7s^3 - 2)(s - 1)(s^2 + s + 1)}
		\end{align*}

	\subsection{}
		\[a^2 + ab + 5a + 5b\]
		\begin{align*}
			&= a(a + b) + 5(a + b) \\
			\Aboxed{{} &= (a + 5)(a + b)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[x^3 - x^2 - x + 1\]
		\begin{align*}
			&= x^2(x - 1) - 1(x - 1) \\
			&= (x^2 - 1)(x - 1) \\
			\Aboxed{{} &= {(x - 1)}^2(x + 1)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[64x^4 + 1\]
		\begin{align*}
			&= 64x^4 + 16x^2 + 1 - 16x^2 \\
			&= 64x^4 + 8x^2 + 8x^2 + 1 - 16x^2 \\
			&= {(8x^2 + 1)}^2 - 16x^2 \\
			\Aboxed{{} &= (8x^2 - 4x + 1)(8x^2 + 4x + 1)}
		\end{align*}
		\hrule{}

	\subsection{}
		\[4m^8 - 4m^4 + 9\]
		\begin{align*}
			&= 4m^8 + 12m^4 + 9 - 16m^4 \\
			&= 4m^8 + 6m^4 + 6m^4 + 9 - 16m^4 \\
			&= {(2m^4 + 3)}^2 - 16m^4 \\
			\Aboxed{{} &= (2m^4 - 4m^2 + 3)(2m^4 + 4m^2 + 3)}
		\end{align*}
