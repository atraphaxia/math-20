	\section{}
		Simplify the following.

		\subsection{}
			\[\frac{7^{-3}}{7^{-16}}\]
			\begin{align*}
				\Aboxed{{} &= 7^{13}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3^2 + 9^{-1} - 9^0}{2^4 - 3^{-1}}\]
			\begin{align*}
				&= \frac{9 + \frac{1}{9} - 1}{16 - \frac{1}{3}} \\
				&= \frac{81 + 1 - 9}{9} \cdot \frac{3}{48 - 1} \\
				&= \frac{73}{3 \cdot 47} \\
				\Aboxed{{} &= \frac{73}{141}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{8x^6y^3z^4}{2x^5yz^2}\]
			\begin{align*}
				\Aboxed{{} &= 4xy^2z^2}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3x^{21}y^{12}z^{19}}{(-x^{-9}y^3z^{-3})(-9x^4y^6z^6)}\]
			\begin{align*}
				&= \frac{3x^{21}y^{12}z^{19}}{9x^{-5}y^9z^3} \\
				\Aboxed{{} &= \frac{1}{3}x^{26}y^3z^{16}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{2{(x^3yz^2)}^2}{{(2x^2y^2z)}^3}\]
			\begin{align*}
				&= \frac{2x^6y^2z^4}{8x^6y^6z^3} \\
				\Aboxed{{} &= \frac{z}{4y^4}}
			\end{align*}

		\subsection{}
			\[\frac{{(2a^3b^{-3}c^{-2})}^3}{{(a^3b^2c^{-1})}^{-2}}\]
			\begin{align*}
				&= \frac{8a^9b^{-9}c^{-6}}{a^{-6}b^{-4}c^2} \\
				\Aboxed{{} &= \frac{8a^{15}}{b^5c^8}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{\left(\frac{2a^3b^{-3}c^{-1}}{3a^{-2}b^{-2}c^3}\right)}^3\]
			\begin{align*}
				&= \frac{8a^9b^{-9}c^{-3}}{27a^{-6}b^{-6}c^9} \\
				\Aboxed{{} &= \frac{8a^{15}}{27b^3c^{12}}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{\left(\frac{a^2b^3c^{-1}}{ab^{-2}c^5}\right)}^4\]
			\begin{align*}
				&= \frac{a^8b^{12}c^{-4}}{a^4b^{-8}c^{20}} \\
				\Aboxed{{} &= \frac{a^4b^{20}}{c^{24}}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{\left(\frac{4^0x^3y^{-2}z^{-4}}{3^2x^{-1}y^3z^{5}}\right)}^{-3}\]
			\begin{align*}
				&= \frac{x^{-9}y^6z^{12}}{3^{-6}x^3y^{-9}z^{15}} \\
				\Aboxed{{} &= \frac{729y^{15}}{x^{12}z^3}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{\left(\frac{18^5 \cdot 20^3x^{-1}y^2z^4}{15^9x^{-3}y^{-2}z^{-1}}\right)}^2\]
			\begin{align*}
				&= \frac{18^{10}20^6x^{-2}y^4z^8}{15^{18}x^{-6}y^{-4}z^{-2}} \\
				&= \frac{2^{10}3^{20}2^{12}5^6x^{-2}y^4z^8}{3^{18}5^{18}x^{-6}y^{-4}z^{-2}} \\
				\Aboxed{{} &= \frac{2^{22}9x^4y^8z^{10}}{5^{12}}}
			\end{align*}

	\section{}
		Factor completely.

		\subsection{}
			\[4x^3y^2 + 6x^2y^3\]
			\begin{align*}
				\Aboxed{{} &= 2x^2y^2(2x + 3y)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[a^2b^3c^4 - a^3b^4c^5 + 2a^2b^4c^4\]
			\begin{align*}
				\Aboxed{{} &= a^2b^3c^4(1 - abc + 2b)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[9a^2 - 1\]
			\begin{align*}
				\Aboxed{{} &= (3a - 1)(3a + 1)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[16x^2 - 25y^2\]
			\begin{align*}
				\Aboxed{{} &= (4x - 5y)(4x + 5y)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[25x^4y^8 - 64z^2\]
			\begin{align*}
				\Aboxed{{} &= (5x^2y^4 - 8z)(5x^2y^4 + 8z)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[16x^4 - 1\]
			\begin{align*}
				&= (4x^2 - 1)(4x^2 + 1) \\
				\Aboxed{{} &= (2x - 1)(2x + 1)(4x^2 + 1)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{(s - 2t)}^2 - 4\]
			\begin{align*}
				\Aboxed{{} &= (s - 2t - 4)(s - 2t + 4)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{(x + 3)}^2 - {(y + 2)}^2\]
			\begin{align*}
				&= (x + 3 - y - 2)(x + 3 + y + 2) \\
				\Aboxed{{} &= {x - y - 1}(x + y + 5)}
			\end{align*}

		\subsection{}
			\[x^2(x^2 - 1) - 9(x^2 - 1)\]
			\begin{align*}
				&= (x^2 - 9)(x^2 - 1) \\
				\Aboxed{{} &= (x - 3)(x + 3)(x - 1)(x + 1)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[8x^9 + z^{12}\]
			\begin{align*}
				\Aboxed{{} &= (2x^3 + z^4)(4x^6 - 2x^3z^4 + z^8)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[8j^3 - 125k^6\]
			\begin{align*}
				\Aboxed{{} &= (2j - 5k^2)(4j^2 + 10jk^2 + 25k^4)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{(x - y)}^3 - 1\]
			\begin{align*}
				\Aboxed{{} &= (x - y - 1)({(x - y)}^2 + (x - y) + 1)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[x^2 - 2x - 3\]
			\begin{align*}
				\Aboxed{{} &= (x - 3)(x + 1)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[s^2 + 7s = 10\]
			\begin{align*}
				\Aboxed{{} &= (s + 5)(s + 2)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[48 - 13q - q^2\]
			\begin{align*}
				&= -(q^2 + 13q - 48) \\
				\Aboxed{{} &= -(q + 16)(q - 3)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[z^6 - 6z^3 - 72\]
			\begin{align*}
				\Aboxed{{} &= (z^3 - 12)(z^3 + 6)}
			\end{align*}

		\subsection{}
			\[t^4 - 9st^2 - 10s^2\]
			\begin{align*}
				\Aboxed{{} &= (t^2 + s)(t^2 - 10s)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[a^6b^4 - a^3b^2c - 42c^2\]
			\begin{align*}
				\Aboxed{{} &= (a^3b^2 - 7c)(a^3b^2 + 6c)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[4x^2 - 4x + 1\]
			\begin{align*}
				&= 4x^2 - 2x - 2x + 1 \\
				\Aboxed{{} &= {(2x - 1)}^2}
			\end{align*}
			\hrule{}

		\subsection{}
			\[4n^2 - 12n + 9\]
			\begin{align*}
				&= 4n^2 - 6n - 6n + 9 \\
				\Aboxed{{} &= {(2n - 3)}^2}
			\end{align*}
			\hrule{}

		\subsection{}
			\[2r^2 - r - 3\]
			\begin{align*}
				&= 2r^2 + 2r - 3r - 3 \\
				\Aboxed{{} &= (2r - 3)(r + 1)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[9a^2 + 15ab + 4b^2\]
			\begin{align*}
				&= 9a^2 + 12ab + 3ab + 4b^2 \\
				\Aboxed{{} &= (3a + b)(3a + 4b)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[7s^6 - 9s^3 + 2\]
			\begin{align*}
				&= 7s^6 - 7s^3 - 2s^3 + 2 \\
				&= (7s^3 - 2)(s^3 - 1) \\
				\Aboxed{{} &= (7s^3 - 2)(s - 1)(s^2 + s + 1)}
			\end{align*}

		\subsection{}
			\[a^2 + ab + 5a + 5b\]
			\begin{align*}
				&= a(a + b) + 5(a + b) \\
				\Aboxed{{} &= (a + 5)(a + b)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[x^3 - x^2 - x + 1\]
			\begin{align*}
				&= x^2(x - 1) - 1(x - 1) \\
				&= (x^2 - 1)(x - 1) \\
				\Aboxed{{} &= {(x - 1)}^2(x + 1)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[64x^4 + 1\]
			\begin{align*}
				&= 64x^4 + 16x^2 + 1 - 16x^2 \\
				&= 64x^4 + 8x^2 + 8x^2 + 1 - 16x^2 \\
				&= {(8x^2 + 1)}^2 - 16x^2 \\
				\Aboxed{{} &= (8x^2 - 4x + 1)(8x^2 + 4x + 1)}
			\end{align*}
			\hrule{}

		\subsection{}
			\[4m^8 - 4m^4 + 9\]
			\begin{align*}
				&= 4m^8 + 12m^4 + 9 - 16m^4 \\
				&= 4m^8 + 6m^4 + 6m^4 + 9 - 16m^4 \\
				&= {(2m^4 + 3)}^2 - 16m^4 \\
				\Aboxed{{} &= (2m^4 - 4m^2 + 3)(2m^4 + 4m^2 + 3)}
			\end{align*}
