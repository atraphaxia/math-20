		\subsection{}
			\[\frac{3a - 3b}{c^2 + 3cd} \cdot \frac{2cd + 6d^2}{ac - bc}\]
			\begin{align*}
				&= \frac{3(a - b)}{c(c + 3d)} \cdot \frac{2d(c + 3d)}{c(a - b)} \\
				\Aboxed{{} &= \frac{6d}{c^2}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{x^3 - 8}{x^2 - 6x + 8} \div \frac{3x^3 + 6x^2 + 12x}{x^2 - 8x + 16}\]
			\begin{align*}
				&= \frac{(x - 2)(x^2 + 2x + 4)}{(x - 4)(x - 2)} \cdot \frac{{(x - 4)}^2}{3x(x^2 + 2x + 4)} \\
				\Aboxed{{} &= \frac{x - 4}{3x}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{x + 1}{x} - \frac{y + 1}{y}\]
			\begin{align*}
				&= \frac{xy + y}{xy} - \frac{xy + x}{xy} \\
				\Aboxed{{} &= \frac{y - x}{xy}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{a}{bc} + \frac{b}{ac} - \frac{c}{ab}\]
			\begin{align*}
				&= \frac{a^2}{abc} + \frac{b^2}{abc} - \frac{c^2}{abc} \\
				\Aboxed{{} &= \frac{a^2 + b^2 - c^2}{abc}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{2}{x^2} + \frac{3}{xy} - \frac{1}{y^2}\]
			\begin{align*}
				&= \frac{2y^2}{x^2y^2} + \frac{3xy}{x^2y^2} - \frac{x^2}{x^2y^2} \\
				\Aboxed{{} &= \frac{2y^2 + 3xy - x^2}{x^2y^2}}
			\end{align*}

		\subsection{}
			\[\frac{1}{x} - \frac{2}{x + y}\]
			\begin{align*}
				&= \frac{x + y}{x(x + y)} - \frac{2x}{x(x + y)} \\
				\Aboxed{{} &= \frac{y - x}{x(x + y)}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{2}{x - y} + \frac{3}{y - x}\]
			\begin{align*}
				&= \frac{3}{y - x} - \frac{2}{y - x} \\
				\Aboxed{{} &= \frac{1}{y - x}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{2}{(a + b)(a - b)} + \frac{1}{{(a - b)}^2}\]
			\begin{align*}
				&= \frac{2a - 2b}{(a + b){(a - b)}^2} + \frac{a + b}{(a + b){(a - b)}^2} \\
				\Aboxed{{} &= \frac{3a - b}{(a + b){(a - b)}^2}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{x^2}{x(x - 1)} + \frac{1}{x} - \frac{1}{x - 1}\]
			\begin{align*}
				&= \frac{x^2}{x(x - 1)} + \frac{x - 1}{x(x - 1)} - \frac{x}{x(x - 1)} \\
				&= \frac{x^2 - 1}{x(x - 1)} \\
				&= \frac{(x - 1)(x + 1)}{x(x - 1)} \\
				\Aboxed{{} &= \frac{x + 1}{x}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{4}{4x^2 - 9} - \frac{3}{2x^2 - x - 3}\]
			\begin{align*}
				&= \frac{4}{(2x - 3)(2x + 3)} - \frac{3}{2x^2 + 2x - 3x - 3} \\
				&= \frac{4}{(2x - 3)(2x + 3)} - \frac{3}{(2x - 3)(x + 1)} \\
				&= \frac{4x + 4}{(2x - 3)(2x + 3)(x + 1)} - \frac{6x + 9}{(2x - 3)(2x + 3)(x + 1)} \\
				\Aboxed{{} &= \frac{-2x - 5}{(2x - 3)(2x + 3)(x + 1)}}
			\end{align*}

		\subsection{}
			\[\frac{3}{2x - 6} - \frac{9}{x^2 - 9}\]
			\begin{align*}
				&= \frac{3}{2(x - 3)} - \frac{9}{(x - 3)(x + 3)} \\
				&= \frac{3x + 9}{2(x - 3)(x + 3)} - \frac{18}{2(x - 3)(x + 3)} \\
				&= \frac{3x - 9}{2(x - 3)(x + 3)} \\
				\Aboxed{{} &= \frac{3}{2(x + 3)}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{x - \frac{9}{x}}{1 - \frac{3}{x}}\]

				&= \frac{2x^2 - 8x + x - 4}{3x^2 - 12x - x + 4} \\
				&= \frac{(2x + 1)(x - 4)}{(3x - 1)(x - 4)} \\
				\Aboxed{{} &= \frac{2x + 1}{3x - 1}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{w - 2 + \frac{w - 2}{w + 2}}{w - \frac{3w + 12}{w + 2}}\]

			\[\frac{a^2 - b^2}{a^3 - b^3}\]
			\begin{align*}

			\hrule{}

		\subsection{}

			\hrule{}
