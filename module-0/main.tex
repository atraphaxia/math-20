	\section{}
		Simplify the following.

		\subsection{}
			\[\frac{7^{-3}}{7^{-16}}\]
			\begin{align*}
				\Aboxed{{} &= 7^{13}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3^2 + 9^{-1} - 9^0}{2^4 - 3^{-1}}\]
			\begin{align*}
				&= \frac{9 + \frac{1}{9} - 1}{16 - \frac{1}{3}} \\
				&= \frac{81 + 1 - 9}{9} \cdot \frac{3}{48 - 1} \\
				&= \frac{73}{3 \cdot 47} \\
				\Aboxed{{} &= \frac{73}{141}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{8x^6y^3z^4}{2x^5yz^2}\]
			\begin{align*}
				\Aboxed{{} &= 4xy^2z^2}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{3x^{21}y^{12}z^{19}}{(-x^{-9}y^3z^{-3})(-9x^4y^6z^6)}\]
			\begin{align*}
				&= \frac{3x^{21}y^{12}z^{19}}{9x^{-5}y^9z^3} \\
				\Aboxed{{} &= \frac{1}{3}x^{26}y^3z^{16}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[\frac{2{(x^3yz^2)}^2}{{(2x^2y^2z)}^3}\]
			\begin{align*}
				&= \frac{2x^6y^2z^4}{8x^6y^6z^3} \\
				\Aboxed{{} &= \frac{z}{4y^4}}
			\end{align*}

		\subsection{}
			\[\frac{{(2a^3b^{-3}c^{-2})}^3}{{(a^3b^2c^{-1})}^{-2}}\]
			\begin{align*}
				&= \frac{8a^9b^{-9}c^{-6}}{a^{-6}b^{-4}c^2} \\
				\Aboxed{{} &= \frac{8a^{15}}{b^5c^8}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{\left(\frac{2a^3b^{-3}c^{-1}}{3a^{-2}b^{-2}c^3}\right)}^3\]
			\begin{align*}
				&= \frac{8a^9b^{-9}c^{-3}}{27a^{-6}b^{-6}c^9} \\
				\Aboxed{{} &= \frac{8a^{15}}{27b^3c^{12}}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{\left(\frac{a^2b^3c^{-1}}{ab^{-2}c^5}\right)}^4\]
			\begin{align*}
				&= \frac{a^8b^{12}c^{-4}}{a^4b^{-8}c^{20}} \\
				\Aboxed{{} &= \frac{a^4b^{20}}{c^{24}}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{\left(\frac{4^0x^3y^{-2}z^{-4}}{3^2x^{-1}y^3z^{5}}\right)}^{-3}\]
			\begin{align*}
				&= \frac{x^{-9}y^6z^{12}}{3^{-6}x^3y^{-9}z^{15}} \\
				\Aboxed{{} &= \frac{729y^{15}}{x^{12}z^3}}
			\end{align*}
			\hrule{}

		\subsection{}
			\[{\left(\frac{18^5 \cdot 20^3x^{-1}y^2z^4}{15^9x^{-3}y^{-2}z^{-1}}\right)}^2\]
			\begin{align*}
				&= \frac{18^{10}20^6x^{-2}y^4z^8}{15^{18}x^{-6}y^{-4}z^{-2}} \\
				&= \frac{2^{10}3^{20}2^{12}5^6x^{-2}y^4z^8}{3^{18}5^{18}x^{-6}y^{-4}z^{-2}} \\
				\Aboxed{{} &= \frac{2^{22}9x^4y^8z^{10}}{5^{12}}}
			\end{align*}
