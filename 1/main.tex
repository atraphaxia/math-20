	\section{} Solve the following equations.
		\subsection{} \[x + 9 = 5 - 3x\]
			\phxproof{
				\phxeqb{}{$x + 3x = 5 - 9$}{Isolate $x$.}
				\phxeqs{}{$4x = -4$}{}
				\phxeqf{}{$x = -1$}{}
			}

		\subsection{} \[\frac{2x + 3}{4} - \frac{x + 1}{2} = -\frac{1}{3}\]
			\phxproof{
				\phxeqb{}{$12(\frac{2x + 3}{4} - \frac{x + 1}{2}) = 12(-\frac{1}{3})$}{$\lcm = 12$.}
				\phxeqs{}{$3(2x + 3) - 6(x + 1) = -4$}{}
				\phxeqs{}{$\cancel{6x} + 9 - \cancel{6x} - 6 = -4$}{}
				\phxeqs{}{$9 - 6 = -4$}{}
				\phxeqs{}{$3 = -4$}{}
				\phxeqf{}{$x \in \emptyset$}{}
			}

		\subsection{} \[3x^2 - 2x + 1 = 0\]
			\phxproof{
				\phxeqb{}{$\frac{-(-2) \pm \sqrt{{(-2)}^2 - 4(3)(1)}}{2(3)}$}{Use the quadratic formula.}
				\phxeqs{}{$\frac{2 \pm \sqrt{4 - 12}}{6}$}{}
				\phxeqs{}{$\frac{2 \pm \sqrt{-8}}{6}$}{}
				\phxeqs{}{$\frac{2 \pm i\sqrt{4}\sqrt{2}}{6}$}{}
				\phxeqs{}{$\frac{2 \pm 2i\sqrt{2}}{6}$}{}
				\phxeqs{}{$\frac{1 \pm i\sqrt{2}}{3}$}{}
			}

		\subsection{} \[4x^2 + 2x = 2\]
			\phxproof{
				\phxeqb{}{$4x^2 + 2x - 2 = 0$}{Rewrite in standard form.}
				\phxeqs{}{$2(2x^2 + x - 1) = 0$}{}
				\phxeqs{}{$(2x - 1)(x + 1) = 0$}{Factor by grouping.}
				\phxeqf{}{$x \in \left\{\frac{1}{2}, -1\right\}$}{}
			}

		\subsection{} \[16x^2 + 9 = 24x\]
			\phxproof{
				\phxeqb{}{$16x^2 - 24x + 9 = 0$}{Rewrite in standard form.}
				\phxeqs{}{$16x^2 - 12x - 12x + 9 = 0$}{Factor by grouping.}
				\phxeqs{}{${(4x - 3)}^2 = 0$}{}
				\phxeqf{}{$x = \frac{3}{4}$}{}
			}

		\subsection{} \[\frac{x}{x - 1} + \frac{x - 5}{x^2 + 2x - 3} = \frac{1}{x + 3}\]
			\phxproof{
				\phxeqb{}{$\frac{x}{x - 1} + \frac{x - 5}{(x - 1)(x + 3)} = \frac{1}{x + 3}$}{Factor by grouping.}
				\phxeqs{}{$\cancel{(x - 1)}\cancel{(x + 3)}(\frac{x}{\cancel{x - 1}} + \frac{x - 5}{\cancel{(x - 1)}\cancel{(x + 3)}}) = (x - 1)\cancel{(x + 3)}(\frac{1}{\cancel{x + 3}})$}{$\lcm = (x - 1)(x + 3)$.}
				\phxeqs{$x \not\in \{1, -3\}$}{$x(x + 3) + x - 5 = x - 1$}{}
				\phxeqs{$x \not\in \{1, -3\}$}{$x^2 + 3x + \cancel{x} - 5 = \cancel{x} - 1$}{}
				\phxeqs{$x \not\in \{1, -3\}$}{$x^2 + 3x - 4 = 0$}{}
				\phxeqs{$x \not\in \{1, -3\}$}{$(x + 4)(x - 1) = 0$}{Factor by grouping.}
				\phxeqs{$x \not\in \{1, -3\}$}{$x \in \{-4, 1\}$}{}
				\phxeqf{}{$x = -4$}{}
			}

		\subsection{} \[\frac{1}{4 - x} + \frac{3}{6 + x} = 0\]
			\phxproof{
				\phxeqb{}{$(4 - x)(6 + x)(\frac{1}{4 - x} + \frac{3}{6 + x}) = 0$}{$\lcm = (4 - x)(6 + x)$.}
				\phxeqs{$x \not\in \{4, -6\}$}{$6 + x + 3(4 - x) = 0$}{}
				\phxeqs{$x \not\in \{4, -6\}$}{$6 + x + 12 - 3x = 0$}{}
				\phxeqs{$x \not\in \{4, -6\}$}{$-2x = -18$}{}
				\phxeqf{}{$x = 9$}{}
			}

		\subsection{} \[\frac{3}{x^2 - 9} - \frac{7}{x - 3} = -\frac{4}{x + 3}\]
			\phxproof{
				\phxeqb{}{$\frac{3}{(x - 3)(x + 3)} - \frac{7}{x - 3} = -\frac{4}{x + 3}$}{Factor using difference of two squares.}
				\phxeqs{}{$\cancel{(x - 3)}\cancel{(x + 3)}(\frac{3}{\cancel{(x - 3)}\cancel{(x + 3)}} - \frac{7}{\cancel{x - 3}}) = \cancel{(x - 3)}\cancel{(x + 3)}(-\frac{4}{\cancel{x + 3}})$}{$lcm = (x - 3)(x + 3)$.}
				\phxeqs{$x \not\in \{3, -3\}$}{$3 - 7(x + 3) = -4(x - 3)$}{}
				\phxeqs{$x \not\in \{3, -3\}$}{$3 - 7x - 21 = -4x + 12$}{}
				\phxeqs{$x \not\in \{3, -3\}$}{$-7x + 4x = 12 + 21 - 3$}{}
				\phxeqs{$x \not\in \{3, -3\}$}{$-3x = 30$}{}
				\phxeqf{}{$x = -10$}{}
			}
