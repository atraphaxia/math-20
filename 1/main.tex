	\section{} Solve the following equations.
		\subsection{} \[x + 9 = 5 - 3x\]
			\phxproof{
				\phxeqb{}{$x + 3x = 5 - 9$}{Isolate $x$.}
				\phxeqs{}{$4x = -4$}{}
				\phxeqf{}{$x = -1$}{}
			}

		\subsection{} \[\frac{2x + 3}{4} - \frac{x + 1}{2} = -\frac{1}{3}\]
			\phxproof{
				\phxeqb{}{$12(\frac{2x + 3}{4} - \frac{x + 1}{2}) = 12(-\frac{1}{3})$}{$\lcm = 12$.}
				\phxeqs{}{$3(2x + 3) - 6(x + 1) = -4$}{}
				\phxeqs{}{$\cancel{6x} + 9 - \cancel{6x} - 6 = -4$}{}
				\phxeqs{}{$9 - 6 = -4$}{}
				\phxeqs{}{$3 = -4$}{}
				\phxeqf{}{$x \in \emptyset$}{}
			}

		\subsection{} \[3x^2 - 2x + 1 = 0\]
			\phxproof{
				\phxeqb{}{$\frac{-(-2) \pm \sqrt{{(-2)}^2 - 4(3)(1)}}{2(3)}$}{Use the quadratic formula.}
				\phxeqs{}{$\frac{2 \pm \sqrt{4 - 12}}{6}$}{}
				\phxeqs{}{$\frac{2 \pm \sqrt{-8}}{6}$}{}
				\phxeqs{}{$\frac{2 \pm i\sqrt{4}\sqrt{2}}{6}$}{}
				\phxeqs{}{$\frac{2 \pm 2i\sqrt{2}}{6}$}{}
				\phxeqs{}{$\frac{1 \pm i\sqrt{2}}{3}$}{}
			}

		\subsection{} \[4x^2 + 2x = 2\]
			\phxproof{
				\phxeqb{}{$4x^2 + 2x - 2 = 0$}{Rewrite in standard form.}
				\phxeqs{}{$2(2x^2 + x - 1) = 0$}{}
				\phxeqs{}{$(2x - 1)(x + 1) = 0$}{Factor by grouping.}
				\phxeqf{}{$x \in \left\{\frac{1}{2}, -1\right\}$}{}
			}

		\subsection{} \[16x^2 + 9 = 24x\]
			\phxproof{
				\phxeqb{}{$16x^2 - 24x + 9 = 0$}{Rewrite in standard form.}
				\phxeqs{}{$16x^2 - 12x - 12x + 9 = 0$}{Factor by grouping.}
				\phxeqs{}{${(4x - 3)}^2 = 0$}{}
				\phxeqf{}{$x = \frac{3}{4}$}{}
			}

		\subsection{} \[\frac{x}{x - 1} + \frac{x - 5}{x^2 + 2x - 3} = \frac{1}{x + 3}\]
			\phxproof{
				\phxeqb{}{$\frac{x}{x - 1} + \frac{x - 5}{(x - 1)(x + 3)} = \frac{1}{x + 3}$}{Factor by grouping.}
				\phxeqs{}{$\cancel{(x - 1)}\cancel{(x + 3)}(\frac{x}{\cancel{x - 1}} + \frac{x - 5}{\cancel{(x - 1)}\cancel{(x + 3)}}) = (x - 1)\cancel{(x + 3)}(\frac{1}{\cancel{x + 3}})$}{$\lcm = (x - 1)(x + 3)$.}
				\phxeqs{$x \not\in \{1, -3\}$}{$x(x + 3) + x - 5 = x - 1$}{}
				\phxeqs{$x \not\in \{1, -3\}$}{$x^2 + 3x + \cancel{x} - 5 = \cancel{x} - 1$}{}
				\phxeqs{$x \not\in \{1, -3\}$}{$x^2 + 3x - 4 = 0$}{}
				\phxeqs{$x \not\in \{1, -3\}$}{$(x + 4)(x - 1) = 0$}{Factor by grouping.}
				\phxeqs{$x \not\in \{1, -3\}$}{$x \in \{-4, 1\}$}{}
				\phxeqf{}{$x = -4$}{}
			}

		\subsection{} \[\frac{1}{4 - x} + \frac{3}{6 + x} = 0\]
			\phxproof{
				\phxeqb{}{$(4 - x)(6 + x)(\frac{1}{4 - x} + \frac{3}{6 + x}) = 0$}{$\lcm = (4 - x)(6 + x)$.}
				\phxeqs{$x \not\in \{4, -6\}$}{$6 + x + 3(4 - x) = 0$}{}
				\phxeqs{$x \not\in \{4, -6\}$}{$6 + x + 12 - 3x = 0$}{}
				\phxeqs{$x \not\in \{4, -6\}$}{$x - 3x = -6 - 12$}{}
				\phxeqs{$x \not\in \{4, -6\}$}{$2x = -18$}{}
				\phxeqf{}{$x = 9$}{}
			}

		\subsection{} \[\frac{3}{x^2 - 9} - \frac{7}{x - 3} = -\frac{4}{x + 3}\]
			\phxproof{
				\phxeqb{}{$\frac{3}{(x - 3)(x + 3)} - \frac{7}{x - 3} = -\frac{4}{x + 3}$}{Factor using difference of two squares.}
				\phxeqs{}{$(x - 3)(x + 3)(\frac{3}{(x - 3)(x + 3)} - \frac{7}{x - 3}) = (x - 3)(x + 3)(-\frac{4}{x + 3})$}{$lcm = (x - 3)(x + 3)$.}
				\phxeqs{$x \not\in \{3, -3\}$}{$3 - 7(x + 3) = -4(x - 3)$}{}
				\phxeqs{$x \not\in \{3, -3\}$}{$3 - 7x - 21 = -4x + 12$}{}
				\phxeqs{$x \not\in \{3, -3\}$}{$-7x + 4x = 12 + 21 - 3$}{}
				\phxeqs{$x \not\in \{3, -3\}$}{$-3x = 30$}{}
				\phxeqf{}{$x = -10$}{}
			}

		\subsection{} \[\frac{3}{x^2 - x - 6} = \frac{4}{2x^2 + x - 6}\]
			\phxproof{
				\phxeqb{}{$\frac{3}{(x - 3)(x + 2)} = \frac{4}{(2x - 3)(x + 2)}$}{Factor by grouping.}
				\phxeqs{}{\scalebox{0.7}{$(x - 3)(2x - 3)(x + 2)(\frac{3}{(x - 3)(x + 2)}) = (x - 3)(2x - 3)(x + 2)(\frac{4}{(2x - 3)(x + 2)})$}}{$lcm = (x - 3)(2x - 3)(x + 2)$.}
				\phxeqs{$x \not\in \left\{3, \frac{3}{2}, -2\right\}$}{$3(2x - 3) = 4(x - 3)$}{}
				\phxeqs{$x \not\in \left\{3, \frac{3}{2}, -2\right\}$}{$6x - 9 = 4x - 12$}{}
				\phxeqs{$x \not\in \left\{3, \frac{3}{2}, -2\right\}$}{$6x - 4x = -12 + 9$}{}
				\phxeqs{$x \not\in \left\{3, \frac{3}{2}, -2\right\}$}{$2x = -3$}{}
				\phxeqf{}{$x = -\frac{3}{2}$}{}
			}

		\subsection{} \[\frac{2}{y + 1} - \frac{3}{1 - y} = \frac{5}{y}\]
			\phxproof{
				\phxeqb{}{$(y + 1)(1 - y)(\frac{2}{y + 1} - \frac{3}{1 - y}) = y(y + 1)(1 - y)(\frac{5}{y})$}{$\lcm = y(y + 1)(1 - y)$.}
				\phxeqs{$y \not\in \{0, -1, 1\}$}{$2y(1 - y) - 3y(y + 1) = 5(y + 1)(1 - y)$}{}
				\phxeqs{$y \not\in \{0, -1, 1\}$}{$2y - 2y^2 - 3y^2 - 3y = 5(1 - y^2)$}{$y + 1 = 1 + y$, expand using difference of two squares.}
				\phxeqs{$y \not\in \{0, -1, 1\}$}{$-2y^2 - 3y^2 - 3y + 2y = 5 - 5y^2$}{}
				\phxeqs{$y \not\in \{0, -1, 1\}$}{$\cancel{-5y^2} - y = 5 - \cancel{5y^2}$}{}
				\phxeqf{}{$y = -5$}{}
			}

		\subsection{} \[\frac{x + 17}{x^2 - 6x + 8} + \frac{x - 2}{x - 4} = \frac{x - 4}{x - 2}\]
			\phxproof{
				\phxeqb{}{$\frac{x + 17}{(x - 2)(x - 4)} + \frac{x - 2}{x - 4} = \frac{x - 4}{x - 2}$}{Factor by grouping.}
				\phxeqs{}{$(x - 2)(x - 4)(\frac{x + 17}{(x - 2)(x - 4)} + \frac{x - 2}{x - 4}) = (x - 2)(x - 4)(\frac{x - 4}{x - 2})$}{$\lcm = (x - 2)(x - 4)$.}
				\phxeqs{$x \not\in \{2, 4\}$}{$x + 17 + {(x - 2)}^2 = {(x - 4)}^2$}{}
				\phxeqs{$x \not\in \{2, 4\}$}{$x + 17 + x^2 - 4x + 4 = x^2 - 8x + 16$}{}
				\phxeqs{$x \not\in \{2, 4\}$}{$\cancel{x^2} - 4x + x + 4 + 17 = \cancel{x^2} - 8x + 16$}{}
				\phxeqs{$x \not\in \{2, 4\}$}{$-3x + 21 = -8x + 16$}{}
				\phxeqs{$x \not\in \{2, 4\}$}{$-3x + 8x = 16 - 21$}{}
				\phxeqs{$x \not\in \{2, 4\}$}{$5x = -5$}{}
				\phxeqf{}{$x = -1$}{}
			}

		\subsection{} \[\frac{w}{3w^2 - 8w + 4} = \frac{w + 2}{3w^2 + w - 2}\]
			\phxproof{
				\phxeqb{}{$\frac{w}{3w^2 - 6w - 2w + 4} = \frac{w + 2}{3w^2 + 3w - 2w - 2}$}{Factor by grouping.}
				\phxeqs{}{$\frac{w}{(3w - 2)(w - 2)} = \frac{w + 2}{(3w - 2)(w + 1)}$}{}
				\phxeqs{}{\scalebox{0.6}{$(3w - 2)(w - 2)(w + 1)(\frac{w}{(3w - 2)(w - 2)}) = (3w - 2)(w - 2)(w + 1)(\frac{w + 2}{(3w - 2)(w + 1)})$}}{$\lcm = (3w - 2)(w - 2)(w + 1)$.}
				\phxeqs{$w \not\in \left\{\frac{2}{3}, 2, -1\right\}$}{$w(w + 1) = (w + 2)(w - 2)$}{}
				\phxeqs{$w \not\in \left\{\frac{2}{3}, 2, -1\right\}$}{$\cancel{w^2} + w = \cancel{w^2} - 4$}{Expand using difference of two squares.}
				\phxeqf{}{$w = -4$}{}
			}

		\subsection{} \[\frac{3}{x + 4} = \frac{2}{3x - 2}\]
			\phxproof{
				\phxeqb{}{$(x + 4)(3x - 2)(\frac{3}{x + 4}) = (x + 4)(3x - 2)(\frac{2}{3x - 2})$}{$\lcm = (x + 4)(3x - 2)$.}
				\phxeqs{$x \not\in \left\{-4, \frac{2}{3}\right\}$}{$3(3x - 2) = 2(x + 4)$}{}
				\phxeqs{$x \not\in \left\{-4, \frac{2}{3}\right\}$}{$9x - 6 = 2x + 8$}{}
				\phxeqs{$x \not\in \left\{-4, \frac{2}{3}\right\}$}{$9x - 2x = 8 + 6$}{}
				\phxeqs{$x \not\in \left\{-4, \frac{2}{3}\right\}$}{$7x = 14$}{}
				\phxeqf{}{$x = 2$}{}
			}

		\subsection{} \[\frac{x + 4}{x + 2} = \frac{2x - 3}{x - 5}\]
			\phxproof{
				\phxeqb{}{$(x + 2)(x - 5)(\frac{x + 4}{x + 2}) = (x + 2)(x - 5)(\frac{2x - 3}{x - 5})$}{$\lcm = (x + 2)(x - 5)$.}
				\phxeqs{$x \not\in \{-2, 5\}$}{$(x + 4)(x - 5) = (2x - 3)(x + 2)$}{}
				\phxeqs{$x \not\in \{-2, 5\}$}{$x^2 - x - 20 = 2x^2 + x - 6$}{}
				\phxeqs{$x \not\in \{-2, 5\}$}{$x^2 - 2x^2 - x - x - 20 + 6 = 0$}{Rewrite in standard form.}
				\phxeqs{$x \not\in \{-2, 5\}$}{$-x^2 - 2x - 14 = 0$}{}
				\phxeqs{$x \not\in \{-2, 5\}$}{$x^2 + 2x + 14 = 0$}{}
				\phxeqs{$x \not\in \{-2, 5\}$}{$\frac{-2 \pm \sqrt{2^2 - 4(1)(14)}}{2(1)}$}{Use the quadratic formula.}
				\phxeqs{$x \not\in \{-2, 5\}$}{$\frac{-2 \pm \sqrt{4 - 56}}{2}$}{}
				\phxeqs{$x \not\in \{-2, 5\}$}{$\frac{-2 \pm \sqrt{-52}}{2}$}{}
				\phxeqs{$x \not\in \{-2, 5\}$}{$\frac{-2 \pm i\sqrt{4}\sqrt{13}}{2}$}{}
				\phxeqs{$x \not\in \{-2, 5\}$}{$\frac{-2 \pm 2i\sqrt{13}}{2}$}{}
				\phxeqf{}{$-1 \pm i\sqrt{13}$}{}
			}

		\subsection{} \[\frac{4}{z - 2} = \frac{z}{6}\]
			\phxproof{
				\phxeqb{}{$6(z - 2)(\frac{4}{z - 2}) = 6(z - 2)(\frac{z}{6})$}{$\lcm = 6(z - 2).$}
				\phxeqs{$z \neq 2$}{$6(4) = z(z - 2)$}{}
				\phxeqs{$z \neq 2$}{$24 = z^2 - 2z$}{}
				\phxeqs{$z \neq 2$}{$z^2 - 2z - 24 = 0$}{Rewrite in standard form.}
				\phxeqs{$z \neq 2$}{$(z - 6)(z + 4) = 0$}{}
				\phxeqf{}{$x \in \{6, -4\}$}{}
			}

		\subsection{} \[\frac{1}{2x + 5} = \frac{4}{2x - 1}\]
			\phxproof{
				\phxeqb{}{$(2x + 5)(2x - 1)(\frac{1}{2x + 5}) = (2x + 5)(2x - 1)(\frac{4}{2x - 1})$}{$\lcm = (2x + 5)(2x - 1)$.}
				\phxeqs{$x \not\in \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}{$2x - 1 = 4(2x + 5)$}{}
				\phxeqs{$x \not\in \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}{$2x - 1 = 8x + 20$}{}
				\phxeqs{$x \not\in \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}{$2x - 8x = 20 + 1$}{}
				\phxeqs{$x \not\in \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}{$-6x = 21$}{}
				\phxeqf{}{$x = -\frac{7}{2}$}{}
			}

		\subsection{} \[\frac{2}{3x} + \frac{4}{5x} = \frac{11}{15}\]
			\phxproof{
				\phxeqb{}{$15x(\frac{2}{3x} + \frac{4}{5x}) = 15x(\frac{11}{15})$}{$\lcm = 15x$.}
				\phxeqs{$x \neq 0$}{$10 + 12 = 11x$}{}
				\phxeqs{$x \neq 0$}{$22 = 11x$}{}
				\phxeqf{}{$x = 2$}{}
			}

		\subsection{} \[\frac{5}{x + 3} = \frac{x + 8}{x + 3} - 3\]
			\phxproof{
				\phxeqb{}{$(x + 3)(\frac{5}{x + 3}) = (x + 3)(\frac{x + 8}{x + 3} - 3)$}{$\lcm = x + 3$.}
				\phxeqs{$x \neq -3$}{$5 = x + 8 - 3(x + 3)$}{}
				\phxeqs{$x \neq -3$}{$5 = x + 8 - 3x - 9$}{}
				\phxeqs{$x \neq -3$}{$5 = -2x - 1$}{}
				\phxeqs{$x \neq -3$}{$2x = -1 - 5$}{}
				\phxeqs{$x \neq -3$}{$2x = -6$}{}
				\phxeqs{$x \neq -3$}{$x = -3$}{}
				\phxeqf{}{$x \in \emptyset$}{}
			}

		\subsection{} \[\frac{3x}{x + 2} = \frac{x + 4}{x}\]
			\phxproof{
				\phxeqb{}{$x(x + 2)(\frac{3x}{x + 2}) = x(x + 2)(\frac{x + 4}{x})$}{$\lcm = x(x + 2)$.}
				\phxeqs{$x \not\in \{0, -2\}$}{$3x^2 = (x + 2)(x + 4)$}{}
				\phxeqs{$x \not\in \{0, -2\}$}{$3x^2 = x^2 + 6x + 8$}{}
				\phxeqs{$x \not\in \{0, -2\}$}{$3x^2 - x^2 - 6x - 8 = 0$}{Rewrite in standard form.}
				\phxeqs{$x \not\in \{0, -2\}$}{$2x^2 - 6x - 8 = 0$}{}
				\phxeqs{$x \not\in \{0, -2\}$}{$x^2 - 3x - 4 = 0$}{}
				\phxeqs{$x \not\in \{0, -2\}$}{$(x - 4)(x + 1) = 0$}{Factor by grouping.}
				\phxeqf{}{$x \in \{4, -1\}$}{}
			}

		\subsection{} \[\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} = \frac{3x - 8}{x^2 - 3x - 10}\]
			\phxproof{
				\phxeqb{}{$\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} = \frac{3x - 8}{(x - 5)(x + 2)}$}{Factor by grouping.}
				\phxeqs{}{$(x - 5)(x + 2)(\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5}) = (x - 5)(x + 2)(\frac{3x - 8}{(x - 5)(x + 2)})$}{$\lcm = (x - 5)(x + 2)$.}
				\phxeqs{$x \not\in \{5, -2\}$}{$(x + 4)(x - 5) - (2x - 3)(x + 2) = 3x - 8$}{}
				\phxeqs{$x \not\in \{5, -2\}$}{$x^2 - x - 20 - (2x^2 + x - 6) = 3x - 8$}{}
				\phxeqs{$x \not\in \{5, -2\}$}{$x^2 - x - 20 - 2x^2 - x + 6 = 3x - 8$}{}
				\phxeqs{$x \not\in \{5, -2\}$}{$x^2 - 2x^2 - x - x - 20 + 6 = 3x - 8$}{}
				\phxeqs{$x \not\in \{5, -2\}$}{$-x^2 - 2x - 14 = 3x - 8$}{}
				\phxeqs{$x \not\in \{5, -2\}$}{$-x^2 - 2x - 3x - 14 + 8 = 0$}{Rewrite in standard form.}
				\phxeqs{$x \not\in \{5, -2\}$}{$-x^2 - 5x - 6 = 0$}{}
				\phxeqs{$x \not\in \{5, -2\}$}{$x^2 + 5x + 6 = 0$}{}
				\phxeqs{$x \not\in \{5, -2\}$}{$(x + 2)(x + 3) = 0$}{Factor by grouping.}
				\phxeqs{$x \not\in \{5, -2\}$}{$x \in \{-2, -3\}$}{}
				\phxeqf{}{$x = -3$}{}
			}

		\subsection{} \[1 + \frac{5x}{6} = \frac{2x^2}{3}\]
			\phxproof{
				\phxeqb{}{$6(1 + \frac{5x}{6}) = 6(\frac{2x^2}{3})$}{$\lcm = 6$.}
				\phxeqs{}{$6 + 5x = 4x^2$}{}
				\phxeqs{}{$4x^2 - 5x - 6 = 0$}{Rewrite in standard form.}
				\phxeqs{}{$4x^2 - 8x + 3x - 6 = 0$}{Factor by grouping.}
				\phxeqs{}{$(4x + 3)(x - 2) = 0$}{}
				\phxeqf{}{$x \in \left\{-\frac{3}{4}, 2\right\}$}{}
			}
