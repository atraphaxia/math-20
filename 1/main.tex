	\section{} Solve the following equations.
		\subsection{} \[x + 9 = 5 - 3x\]
			\njphxproof{
				\njphxeqb{}{$x + 3x = 5 - 9$}{Solve for $x$.}
				\njphxeqs{}{$4x = -4$}{}
				\njphxeqf{}{$x = -1$}{}
			}

		\subsection{} \[\frac{2x + 3}{4} - \frac{x + 1}{2} = -\frac{1}{3}\]
			\njphxproof{
				\njphxeqb{}{$\frac{3(2x + 3)}{12} - \frac{6(x + 1)}{12} = -\frac{4}{12}$}{$\lcm = 12$.}
				\njphxeqs{}{$3(2x + 3) - 6(x + 1) = -4$}{}
				\njphxeqs{}{$6x + 9 - 6x - 6 = -4$}{}
				\njphxeqs{}{$3 = -4$}{}
				\njphxeqf{}{$x \in \emptyset$}{}
			}

		\subsection{} \[3x^2 - 2x + 1 = 0\]
			\njphxproof{
				\njphxeqb{}{$x = \frac{-(-2) \pm \sqrt{{(-2)}^2 - 4(3)(1)}}{2(3)}$}{Use the quadratic formula.}
				\njphxeqs{}{$x = \frac{2 \pm \sqrt{4 - 12}}{6}$}{}
				\njphxeqs{}{$x = \frac{2 \pm \sqrt{-8}}{6}$}{}
				\njphxeqs{}{$x = \frac{2 \pm i\sqrt{4}\sqrt{2}}{6}$}{}
				\njphxeqs{}{$x = \frac{2 \pm 2i\sqrt{2}}{6}$}{}
				\njphxeqf{}{$x = \frac{1 \pm i\sqrt{2}}{3}$}{Simplify.}
			}

		\subsection{} \[4x^2 + 2x = 2\]
			\njphxproof{
				\njphxeqb{}{$4x^2 + 2x - 2 = 0$}{Rewrite in standard form.}
				\njphxeqs{}{$4x^2 + 4x - 2x - 2 = 0$}{Factor.}
				\njphxeqs{}{$4x(x + 1) - 2(x + 1) = 0$}{}
				\njphxeqs{}{$(4x - 2)(x + 1) = 0$}{}
				\njphxeqs{}{$2(2x - 1)(x + 1) = 0$}{}
				\njphxeqf{}{$x \in \left\{\frac{1}{2}, -1\right\}$}{}
			}

		\subsection{} \[16x^2 + 9 = 24x\]
			\njphxproof{
				\njphxeqb{}{$16x^2 - 24x + 9 = 0$}{Rewrite in standard form.}
				\njphxeqs{}{$16x^2 - 12x - 12x + 9 = 0$}{Factor.}
				\njphxeqs{}{$4x(4x - 3) - 3(4x - 3) = 0$}{}
				\njphxeqs{}{${(4x - 3)}^2 = 0$}{}
				\njphxeqf{}{$x = \frac{3}{4}$}{}
			}

		\subsection{} \[\frac{x}{x - 1} + \frac{x - 5}{x^2 + 2x - 3} = \frac{1}{x + 3}\]
			\njphxproof{
				\njphxeqb{}{$\frac{x}{x - 1} + \frac{x - 5}{(x - 1)(x + 3)} = \frac{1}{x + 3}$}{Factor.}
				\njphxeqs{}{$\frac{x(x + 3)}{(x - 1)(x + 3)} + \frac{x - 5}{(x - 1)(x + 3)} = \frac{x - 1}{(x - 1)(x + 3)}$}{$\lcm = (x - 1)(x + 3)$.}
				\njphxeqs{$x \not\in \{1, -3\}$}{$x(x + 3) + x - 5 = x - 1$}{Cancel denominators.}
				\njphxeqs{$x \not\in \{1, -3\}$}{$x^2 + 3x + x - 5 = x - 1$}{}
				\njphxeqs{$x \not\in \{1, -3\}$}{$x^2 + 3x + x - 5 - x + 1 = 0$}{Rewrite in standard form.}
				\njphxeqs{$x \not\in \{1, -3\}$}{$x^2 + 3x - 4 = 0$}{}
				\njphxeqs{$x \not\in \{1, -3\}$}{$(x + 4)(x - 1) = 0$}{Factor.}
				\njphxeqs{$x \not\in \{1, -3\}$}{$x \in \{-4, 1\}$}{}
				\njphxeqf{}{$x = -4$}{}
			}

		\subsection{} \[\frac{1}{4 - x} + \frac{3}{6 + x} = 0\]
			\njphxproof{
				\njphxeqb{}{$\frac{6 + x}{(4 - x)(6 + x)} + \frac{3(4 - x)}{(4 - x)(6 + x)} = 0$}{$\lcm = (4 - x)(6 + x)$}
				\njphxeqs{$x \not\in \{4, -6\}$}{$6 + x + 3(4 - x) = 0$}{Cancel denominators.}
				\njphxeqs{$x \not\in \{4, -6\}$}{$6 + x + 12 - 3x = 0$}{}
				\njphxeqs{$x \not\in \{4, -6\}$}{$-2x = -18$}{}
				\njphxeqf{}{$x = 9$}{}
			}

		\subsection{} \[\frac{3}{x^2 - 9} - \frac{7}{x - 3} = -\frac{4}{x + 3}\]
			\njphxproof{
				\njphxeqb{}{$\frac{3}{(x - 3)(x + 3)} - \frac{7}{x - 3} = -\frac{4}{x + 3}$}{Factor using difference of two squares.}
				\njphxeqs{}{$\frac{3}{(x - 3)(x + 3)} - \frac{7(x + 3)}{(x - 3)(x + 3)} = -\frac{4(x - 3)}{(x - 3)(x + 3)}$}{$\lcm = (x - 3)(x + 3)$}
				\njphxeqs{$x \not\in \{-3, 3\}$}{$3 - 7(x + 3) = -4(x - 3)$}{Cancel denominators.}
				\njphxeqs{$x \not\in \{-3, 3\}$}{$3 - 7x - 21 = -4x + 12$}{}
				\njphxeqs{$x \not\in \{-3, 3\}$}{$-3x = 30$}{}
				\njphxeqf{}{$x = -10$}{}
			}
