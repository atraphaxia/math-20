\subsection{} \[1 + \frac{5x}{6} = \frac{2x^2}{3}\]
	\phxproof{
		\phxeqb{}{$6\left(1 + \frac{5x}{6}\right) = 6\left(\frac{2x^2}{3}\right)$}{$\lcm = 6$.}
		\phxeqs{}{$6 + 5x = 4x^2$}{}
		\phxeqs{}{$4x^2 - 5x - 6 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$4x^2 - 8x + 3x - 6 = 0$}{Factor by grouping.}
		\phxeqs{}{$(4x + 3)(x - 2) = 0$}{}
		\phxeqf{}{$x \in \left\{-\frac{3}{4}, 2\right\}$}{}
	}

\subsection{} \[\frac{x}{2x + 3} = x + 4\]
	\phxproof{
		\phxeqb{$x \neq -\frac{3}{2}$}{$x = (2x + 3)(x + 4)$}{Eliminate denominator.}
		\phxeqs{$x \neq -\frac{3}{2}$}{$x = 2x^2 + 11x + 12$}{}
		\phxeqs{$x \neq -\frac{3}{2}$}{$2x^2 + 11x - x + 12 = 0$}{Rewrite in standard form.}
		\phxeqs{$x \neq -\frac{3}{2}$}{$2x^2 + 10x + 12 = 0$}{}
		\phxeqs{$x \neq -\frac{3}{2}$}{$x^2 + 5x + 6 = 0$}{}
		\phxeqs{$x \neq -\frac{3}{2}$}{$(x + 3)(x + 2) = 0$}{Factor by grouping.}
		\phxeqf{}{$x \in \{-3, -2\}$}{}
	}

\subsection{} \[\sqrt{x} + 5 = 3\]
	\phxproof{
		\phxeqb{}{$\sqrt{x} = 3 - 5$}{Isolate radical.}
		\phxeqs{}{$\sqrt{x} = -2$}{}
		\phxeqs{}{$x = 4$}{Eliminate radical.}
		\phxeqb{}{$\sqrt{4} + 5 = 3$}{Check $x = 4$.}
		\phxeqs{}{$2 + 5 = 3$}{}
		\phxeqs{}{$7 \neq 3$}{}
		\phxeqf{}{$x \in \emptyset$}{}
	}

\subsection{} \[\sqrt{3x + 7} - x = 1\]
	\phxproof{
		\phxeqb{}{$\sqrt{3x + 7} = x + 1$}{Isolate radical.}
		\phxeqs{}{$3x + 7 = {(x + 1)}^2$}{Eliminate radical.}
		\phxeqs{}{$3x + 7 = x^2 + 2x + 1$}{}
		\phxeqs{}{$x^2 + 2x - 3x + 1 - 7 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$x^2 - x - 6 = 0$}{}
		\phxeqs{}{$(x - 3)(x + 2) = 0$}{Factor by grouping.}
		\phxeqs{}{$x \in \{3, -2\}$}{}
		\phxeqb{}{$\sqrt{3(3) + 7} = 3 + 1$}{Check $x = 3$.}
		\phxeqs{}{$\sqrt{9 + 7} = 3 + 1$}{}
		\phxeqs{}{$\sqrt{16} = 4$}{}
		\phxeqs{}{$4 = 4$}{}
		\phxeqb{}{$\sqrt{3(-2) + 7} = 3 + 1$}{Check $x = -2$.}
		\phxeqs{}{$\sqrt{-6 + 7} = 3 + 1$}{}
		\phxeqs{}{$\sqrt{1} = 4$}{}
		\phxeqs{}{$1 \neq 4$}{}
		\phxeqb{}{$x = 3$}{}
	}

\subsection{} \[\sqrt{\sqrt{5x - 1} - x} = 1\]
	\phxproof{
		\phxeqb{}{$\sqrt{5x - 1} - x = 1$}{Eliminate radical.}
		\phxeqs{}{$\sqrt{5x - 1} = x + 1$}{Isolate radical.}
		\phxeqs{}{$5x - 1 = {(x + 1)}^2$}{Eliminate radical.}
		\phxeqs{}{$5x - 1 = x^2 + 2x + 1$}{}
		\phxeqs{}{$x^2 + 2x - 5x + 1 + 1 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$x^2 - 3x + 2 = 0$}{}
		\phxeqs{}{$(x - 2)(x - 1) = 0$}{Factor by grouping.}
		\phxeqs{}{$x \in \{2, 1\}$}{}
		\phxeqb{}{$\sqrt{\sqrt{5(2) - 1} - 2} = 1$}{Check $x = 2$.}
		\phxeqs{}{$\sqrt{\sqrt{9} - 2} = 1$}{}
		\phxeqs{}{$\sqrt{1} = 1$}{}
		\phxeqs{}{$1 = 1$}{}
		\phxeqb{}{$\sqrt{\sqrt{5(1) - 1} - 1} = 1$}{Check $x - 1$.}
		\phxeqs{}{$\sqrt{\sqrt{4} - 1} = 1$}{}
		\phxeqs{}{$\sqrt{1} = 1$}{}
		\phxeqs{}{$1 = 1$}{}
		\phxeqf{}{$x \in \{2, 1\}$}{}
	}

\subsection{} \[\sqrt{5w + 1} - \sqrt{3w} - 1 = 0\]
	\phxproof{
		\phxeqb{}{$\sqrt{5w + 1} - 1 = \sqrt{3w}$}{Eliminate radicals.}
		\phxeqs{}{$5w + 1 - 2\sqrt{5w + 1} + 1 = 3w$}{}
		\phxeqs{}{$5w - 3w + 1 + 1 = 2\sqrt{5w + 1}$}{}
		\phxeqs{}{$2w + 2 = 2\sqrt{5w + 1}$}{}
		\phxeqs{}{${(2w + 2)}^2 = 4(5w + 1)$}{}
		\phxeqs{}{$4w^2 + 8w + 4 = 20w + 4$}{Rewrite in standard form.}
		\phxeqs{}{$4w^2 + 8w - 20w + 4 - 4 = 0$}{}
		\phxeqs{}{$4w^2 - 12w = 0$}{}
		\phxeqs{}{$4w(w - 3) = 0$}{}
		\phxeqs{}{$w \in \{0, 3\}$}{}
		\phxeqb{}{$\sqrt{5(0) + 1} - \sqrt{3(0)} - 1 = 0$}{Check $x = 0$.}
		\phxeqs{}{$\sqrt{1} - \sqrt{0} - 1 = 0$}{}
		\phxeqs{}{$0 = 0$}{}
		\phxeqb{}{$\sqrt{5(3) + 1} - \sqrt{3(3)} - 1 = 0$}{Check $x = 3$.}
		\phxeqs{}{$\sqrt{16} - \sqrt{9} - 1 = 0$}{}
		\phxeqs{}{$0 = 0$}{}
		\phxeqf{}{$w \in \{0, 3\}$}{}
	}

\subsection{} \[\sqrt{x - 2} = 2\]
	\phxproof{
		\phxeqb{}{$x - 2 = 4$}{Eliminate radical.}
		\phxeqs{}{$x = 6$}{}
		\phxeqb{}{$\sqrt{6 - 2} = 2$}{Check $x = 6$.}
		\phxeqs{}{$2 = 2$}{}
		\phxeqf{}{$x = 6$}{}
	}

\subsection{} \[\sqrt{2x + 5} + x = 5\]
	\phxproof{
		\phxeqb{}{$\sqrt{2x + 5} = 5 - x$}{Isolate radical.}
		\phxeqs{}{$2x + 5 = {(5 - x)}^2$}{Eliminate radical.}
		\phxeqs{}{$2x + 5 = 25 - 10x + x^2$}{}
		\phxeqs{}{$x^2 - 10x - 2x + 25 - 5 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$x^2 - 12x + 20 = 0$}{}
		\phxeqs{}{$(x - 10)(x - 2) = 0$}{Factor by grouping.}
		\phxeqs{}{$x \in \{10, 2\}$}{}
		\phxeqb{}{$\sqrt{2(10) + 5} + 10 = 5$}{Check $x = 10$.}
		\phxeqs{}{$\sqrt{20 + 5} + 10 = 5$}{}
		\phxeqs{}{$15 \neq 5$}{}
		\phxeqb{}{$\sqrt{2(2) + 5} + 2 = 5$}{Check $x = 2$.}
		\phxeqs{}{$\sqrt{4 + 5} + 2 = 5$}{}
		\phxeqs{}{$5 = 5$}{}
		\phxeqf{}{$x = 2$}{}
	}

\subsection{} \[\sqrt{x^2 - 8x} = 3\]
	\phxproof{
		\phxeqb{}{$x^2 - 8x = 9$}{Eliminate radical.}
		\phxeqs{}{$x^2 - 8x - 9 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$(x + 1)(x - 9) = 0$}{}
		\phxeqs{}{$x \in \{-1, 9\}$}{}
		\phxeqb{}{$\sqrt{{(-1)}^2 - 8(-1)} = 3$}{Check $x = -1$.}
		\phxeqs{}{$\sqrt{1 + 8} = 3$}{}
		\phxeqs{}{$3 = 3$}{}
		\phxeqb{}{$\sqrt{9^2 - 8(9)} = 3$}{Check $x = 9$.}
		\phxeqs{}{$\sqrt{81 - 72} = 3$}{}
		\phxeqs{}{$3 = 3$}{}
		\phxeqf{}{$x \in \{-1, 9\}$}{}
	}

\subsection{} \[\sqrt{x + \sqrt{x - 2}} = 2\]
	\phxproof{
		\phxeqb{}{$x + \sqrt{x - 2} = 4$}{Eliminate radical.}
		\phxeqs{}{$\sqrt{x - 2} = 4 - x$}{Isolate radical.}
		\phxeqs{}{$x - 2 = {(4 - x)}^2$}{Eliminate radical.}
		\phxeqs{}{$x - 2 = 16 - 8x + x^2$}{}
		\phxeqs{}{$x^2 - 8x - x + 16 + 2 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$x^2 - 9x + 18 = 0$}{}
		\phxeqs{}{$(x - 6)(x - 3) = 0$}{}
		\phxeqs{}{$x \in \{6, 3\}$}{}
		\phxeqb{}{$\sqrt{6 + \sqrt{6 - 2}} = 2$}{Check $x = 6$.}
		\phxeqs{}{$\sqrt{6 + 2} = 2$}{}
		\phxeqs{}{$\sqrt{8} \neq 2$}{}
		\phxeqb{}{$\sqrt{3 + \sqrt{3 - 2}} = 2$}{Check $x = 3$.}
		\phxeqs{}{$\sqrt{3 + 1} = 2$}
		\phxeqs{}{$2 = 2$}
		\phxeqf{}{$x = 3$}{}
	}
