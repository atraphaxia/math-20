\subsection{} \[\frac{15}{t^2} - \frac{14}{t} - 8 = 0\]
	\phxproof{
		\phxeqb{}{$t^2(\frac{15}{t^2} - \frac{14}{t} - 8) = t^2(0)$}{$\lcm = t^2$.}
		\phxeqs{$t \ne 0$}{$15 - 14t - 8t^2 = 0$}{}
		\phxeqs{$t \ne 0$}{$8t^2 + 14t - 15 = 0$}{Rewrite in standard form.}
		\phxeqs{$t \ne 0$}{$t = \frac{-14 \pm \sqrt{14^2 - 4(8)(-15)}}{2(8)}$}{Use the quadratic formula.}
		\phxeqs{$t \ne 0$}{$t = \frac{-14 \pm \sqrt{196 + 480}}{16}$}{}
		\phxeqs{$t \ne 0$}{$t = \frac{-14 \pm \sqrt{676}}{16}$}{}
		\phxeqs{$t \ne 0$}{$t = \frac{-14 \pm 26}{16}$}{}
		\phxeqs{$t \ne 0$}{$t \in \{\frac{-14 + 26}{16}, \frac{-14 + 26}{16}\}$}{}
		\phxeqs{$t \ne 0$}{$t \in \{\frac{12}{16}, -\frac{40}{16}\}$}{}
		\phxeqf{}{$t \in \{\frac{3}{4}, -\frac{5}{2}\}$}{}
	}

\subsection{} \[t^4 - 4 = 0\]
	\phxproof{
		\phxeqb{}{$(t^2 - 2)(t^2 + 2) = 0$}{Factor using difference of two squares.}
		\phxeqs{}{$(t - \sqrt{2})(t + \sqrt{2})(t^2 + 2) = 0$}{Factor using difference of two squares. $\pm \sqrt{2}$ are solutions.}
		\phxeqs{$t = \pm \sqrt{2}$}{$t^2 + 2 = 0$}{Find the zeroes of $t^2 + 2$.}
		\phxeqs{$t = \pm \sqrt{2}$}{$t^2 = -2$}{}
		\phxeqs{$t = \pm \sqrt{2}$}{$t = \pm \sqrt{-2}$}{}
		\phxeqs{$t = \pm \sqrt{2}$}{$t = \pm i\sqrt{2}$}{}
		\phxeqf{}{$t \in \{\pm \sqrt{2}, \pm i\sqrt{2}\}$}{}
	}

\subsection{} \[x^4 - 2x^2 - 15 = 0\]
	\phxproof{
		\phxeqb{}{$t^2 - 2t - 15 = 0$}{$t = x^2$.}
		\phxeqs{}{$(t - 5)(t + 3) = 0$}{Factor by grouping.}
		\phxeqs{}{$t \in \{5, 3\}$}{}
		\phxeqb{}{$5 = x^2$}{Solve for $x$, $t = 5$.}
		\phxeqs{}{$x = \pm \sqrt{5}$}{}
		\phxeqb{}{$-3 = x^2$}{Solve for $x$, $t = -3$.}
		\phxeqs{}{$x = \pm i\sqrt{3}$}{}
		\phxeqf{}{$x \in \{\pm \sqrt{5}, \pm i\sqrt{3}\}$}{}
	}

\subsection{} \[2{(2x + \frac{3}{2})}^2 - 7(2x + \frac{3}{2}) + 3 = 0\]
	\phxproof{
		\phxeqb{}{$2t^2 - 7t + 3 = 0$}{$t = 2x + \frac{3}{2}$.}
		\phxeqs{}{$2t^2 - 6t - t + 3 = 0$}{Factor by grouping.}
		\phxeqs{}{$(2t - 1)(t - 3) = 0$}{}
		\phxeqs{}{$t \in \{\frac{1}{2}, 3\}$}{}
		\phxeqb{}{$\frac{1}{2} = 2x + \frac{3}{2}$}{Solve for $x$, $t = \frac{1}{2}$.}
		\phxeqs{}{$\frac{1}{2} - \frac{3}{2} = 2x$}{}
		\phxeqs{}{$-1 = 2x$}{}
		\phxeqs{}{$x = -\frac{1}{2}$}{}
		\phxeqb{}{$3 = 2x + \frac{3}{2}$}{Solve for $x$, $t = 3$.}
		\phxeqs{}{$3 - \frac{3}{2} = 2x$}{}
		\phxeqs{}{$\frac{3}{2} = 2x$}{}
		\phxeqs{}{$x = \frac{3}{4}$}{}
		\phxeqf{}{$x \in \{-\frac{1}{2}, \frac{3}{4}\}$}{}
	}

\subsection{} \[2x^{\frac{2}{3}} - 5x^{\frac{1}{3}} - 3 = 0\]
	\phxproof{
		\phxeqb{}{$2t^2 - 5t - 3 = 0$}{$t = x^{\frac{1}{3}}$.}
		\phxeqs{}{$2t^2 - 6t + t - 3 = 0$}{Factor by grouping.}
		\phxeqs{}{$(2t + 1)(t - 3) = 0$}{}
		\phxeqs{}{$t \in \{-\frac{1}{2}, 3\}$}{}
		\phxeqb{}{$-\frac{1}{2} = x^{\frac{1}{3}}$}{Solve for $x$, $t = -\frac{1}{2}$.}
		\phxeqs{}{${(-\frac{1}{2})}^3 = x$}{}
		\phxeqs{}{$x = -\frac{1}{8}$}{}
		\phxeqb{}{$3 = x^{\frac{1}{3}}$}{Solve for $x$, $t = 3$.}
		\phxeqs{}{$3^3 = x$}{}
		\phxeqs{}{$x = 27$}{}
		\phxeqf{}{$x \in \{-\frac{1}{8}, 27\}$}{}
	}

\subsection{} \[3{(4x - \frac{1}{4})}^2 - 4(4x - \frac{1}{4}) - 15 = 0\]
	\phxproof{
		\phxeqb{}{$3t^2 - 4t - 15 = 0$}{$t = 4x - \frac{1}{4}$}
		\phxeqs{}{$3t^2 - 9t + 5t - 15 = 0$}{Factor by grouping.}
		\phxeqs{}{$(3t + 5)(t - 3) = 0$}{}
		\phxeqs{}{$t \in \{-\frac{5}{3}, 3\}$}{}
		\phxeqb{}{$-\frac{5}{3} = 4x - \frac{1}{4}$}{Solve for $x$, $t = -\frac{5}{3}$.}
		\phxeqs{}{$-\frac{5}{3} + \frac{1}{4} = 4x$}{}
		\phxeqs{}{$-\frac{20}{12} + \frac{3}{12} = 4x$}{}
		\phxeqs{}{$-\frac{17}{12} = 4x$}{}
		\phxeqs{}{$x = -\frac{17}{48}$}{}
		\phxeqb{}{$3 = 4x - \frac{1}{4}$}{Solve for $x$, $t = 3$.}
		\phxeqs{}{$3 + \frac{1}{4} = 4x$}{}
		\phxeqs{}{$\frac{13}{4} = 4x$}{}
		\phxeqs{}{$x = \frac{13}{16}$}{}
		\phxeqf{}{$x \in \{-\frac{17}{48}, \frac{13}{16}\}$}{}
	}

\subsection{} \[6 + \sqrt[3]{x + 1} = \sqrt[3]{{(x + 1)}^2}\]
	\phxproof{
		\phxeqb{}{$6 + t = t^2$}{$t = \sqrt[3]{x + 1}$}
		\phxeqs{}{$t^2 - t - 6 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$(t - 3)(t + 2) = 0$}{Factor by grouping.}
		\phxeqs{}{$t \in \{3, -2\}$}{}
		\phxeqb{}{$3 = \sqrt[3]{x + 1}$}{Solve for $x$, $t = 3$.}
		\phxeqs{}{$3^3 = x + 1$}{Eliminate radical.}
		\phxeqs{}{$27 = x + 1$}{}
		\phxeqs{}{$x = 26$}{}
		\phxeqb{}{$3 = \sqrt[3]{26 + 1}$}{Check $x = 26$.}
		\phxeqs{}{$3 = 3$}{}
		\phxeqb{}{$-2 = \sqrt[3]{x + 1}$}{Solve for $x$, $t = -2$.}
		\phxeqs{}{$(-2)^3 = x + 1$}{Eliminate radical.}
		\phxeqs{}{$-8 = x + 1$}{}
		\phxeqs{}{$x = -9$}{}
		\phxeqb{}{$-2 = \sqrt[3]{-9 + 1}$}{Check $x = -9$.}
		\phxeqs{}{$-2 = -2$}{}
		\phxeqf{}{$x \in \{26, -9\}$}{}
	}

\subsection{} \[-7(\frac{1}{x} - 1) = 4 - 2{(\frac{1}{x} - 1)}^2\]
	\phxproof{
		\phxeqb{}{$-7t = 4 - 2t^2$}{$t = \frac{1}{x} - 1$}
		\phxeqs{}{$2t^2 - 7t - 4 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$2t^2 - 8t + t - 4 = 0$}{Factor by grouping.}
		\phxeqs{}{$(2t + 1)(t - 4) = 0$}{}
		\phxeqs{}{$t \in \{-\frac{1}{2}, 4\}$}{}
		\phxeqb{}{$-\frac{1}{2} = \frac{1}{x} - 1$}{Solve for $x$, $t = -\frac{1}{2}$.}
		\phxeqs{}{$2x(-\frac{1}{2}) = 2x(\frac{1}{x} - 1)$}{$\lcm = 2x$.}
		\phxeqs{$x \ne 0$}{$-x = 2 - 2x$}{}
		\phxeqs{}{$x = 2$}{}
		\phxeqb{}{$4 = \frac{1}{x} - 1$}{Solve for $x$, $t = 4$.}
		\phxeqs{$x \ne 0$}{$4x = 1 - x$}{$\lcm = x$.}
		\phxeqs{$x \ne 0$}{$5x = 1$}{}
		\phxeqs{}{$x = \frac{1}{5}$}{}
		\phxeqf{}{$x \in \{2, \frac{1}{5}\}$}{}
	}

\subsection{} \[x^2(x^2 - 1) - 9(x^2 - 1) = 0\]
	\phxproof{
		\phxeqb{}{$x^4 - x^2 - 9x^2 + 9 = 0$}{}
		\phxeqs{}{$x^4 - 10x^2 + 9 = 0$}{}
		\phxeqs{}{$t^2 - 10t + 9 = 0$}{$t = x^2$.}
		\phxeqs{}{$(t - 9)(t - 1) = 0$}{Factor by grouping.}
		\phxeqs{}{$t \in \{9, 1\}$}{}
		\phxeqb{}{$9 = x^2$}{Solve for $x$, $t = 9$.}
		\phxeqs{}{$x = \pm 3$}{}
		\phxeqb{}{$1 = x^2$}{Solve for $x$, $t = 1$.}
		\phxeqs{}{$x = \pm 1$}{}
		\phxeqf{}{$x \in \{\pm 3, \pm 1\}$}{}
	}

\subsection{} \[2(x^2 + x + 1) + \sqrt{x^2 + x + 1} - 3 = 0\]
	\phxproof{
		\phxeqb{}{$2t^2 + t - 3 = 0$}{$t = \sqrt{x^2 + x + 1}$.}
		\phxeqs{}{$2t^2 + 3t - 2t - 3 = 0$}{Factor by grouping.}
		\phxeqs{}{$(t - 1)(2t + 3) = 0$}{}
		\phxeqs{}{$t \in \{1, -\frac{3}{2}\}$}{}
		\phxeqb{}{$1 = \sqrt{x^2 + x + 1}$}{Solve for $x$, $t = 1$.}
		\phxeqs{}{$1 = x^2 + x + 1$}{Eliminate radical.}
		\phxeqs{}{$x^2 + x + 1 - 1 = 0$}{Rewrite in standard form.}
		\phxeqs{}{$x^2 + x = 0$}{}
		\phxeqs{}{$x(x + 1) = 0$}{}
		\phxeqs{}{$x \in \{0, -1\}$}{}
		\phxeqb{}{$1 = \sqrt{0^2 + 0 + 1}$}{Check $x = 0$.}
		\phxeqs{}{$1 = 1$}{}
		\phxeqb{}{$1 = \sqrt{{(-1)}^2 + (-1) + 1}$}{Check $x = -1$.}
		\phxeqs{}{$1 = 1$}{}
		\phxeqb{}{$-\frac{3}{2} = \sqrt{x^2 + x + 1}$}{Solve for $x$, $t = -\frac{3}{2}$.}
		\phxeqs{}{$x \in \emptyset$}{The square root of a real number cannot be negative.}
		\phxeqf{}{$x \in \{0, 1\}$}{}
	}
