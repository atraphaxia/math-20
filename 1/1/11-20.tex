\subsection{} \[\frac{x + 17}{x^2 - 6x + 8} + \frac{x - 2}{x - 4} = \frac{x - 4}{x - 2}\]
	\phxproof{
		\phxeqb{}{$\frac{x + 17}{(x - 2)(x - 4)} + \frac{x - 2}{x - 4} = \frac{x - 4}{x - 2}$}{Factor by grouping.}
		\phxeqs{}{$(x - 2)(x - 4)\left(\frac{x + 17}{(x - 2)(x - 4)} + \frac{x - 2}{x - 4}\right) = (x - 2)(x - 4)\left(\frac{x - 4}{x - 2}\right)$}{$\lcm = (x - 2)(x - 4)$.}
		\phxeqs{$x \not\in \{2, 4\}$}{$x + 17 + {(x - 2)}^2 = {(x - 4)}^2$}{}
		\phxeqs{$x \not\in \{2, 4\}$}{$x + 17 + x^2 - 4x + 4 = x^2 - 8x + 16$}{}
		\phxeqs{$x \not\in \{2, 4\}$}{$\cancel{x^2} - 4x + x + 4 + 17 = \cancel{x^2} - 8x + 16$}{}
		\phxeqs{$x \not\in \{2, 4\}$}{$-3x + 21 = -8x + 16$}{}
		\phxeqs{$x \not\in \{2, 4\}$}{$-3x + 8x = 16 - 21$}{}
		\phxeqs{$x \not\in \{2, 4\}$}{$5x = -5$}{}
		\phxeqf{}{$x = -1$}{}
	}

\subsection{} \[\frac{w}{3w^2 - 8w + 4} = \frac{w + 2}{3w^2 + w - 2}\]
	\phxproof{
		\phxeqb{}{$\frac{w}{3w^2 - 6w - 2w + 4} = \frac{w + 2}{3w^2 + 3w - 2w - 2}$}{Factor by grouping.}
		\phxeqs{}{$\frac{w}{(3w - 2)(w - 2)} = \frac{w + 2}{(3w - 2)(w + 1)}$}{}
		\phxeqs{}{\scalebox{0.6}{$(3w - 2)(w - 2)(w + 1)\left(\frac{w}{(3w - 2)(w - 2)}\right) = (3w - 2)(w - 2)(w + 1)\left(\frac{w + 2}{(3w - 2)(w + 1)}\right)$}}{$\lcm = (3w - 2)(w - 2)(w + 1)$.}
		\phxeqs{$w \not\in \left\{\frac{2}{3}, 2, -1\right\}$}{$w(w + 1) = (w + 2)(w - 2)$}{}
		\phxeqs{$w \not\in \left\{\frac{2}{3}, 2, -1\right\}$}{$\cancel{w^2} + w = \cancel{w^2} - 4$}{Expand using difference of two squares.}
		\phxeqf{}{$w = -4$}{}
	}

\subsection{} \[\frac{3}{x + 4} = \frac{2}{3x - 2}\]
	\phxproof{
		\phxeqb{}{$(x + 4)(3x - 2)\left(\frac{3}{x + 4}\right) = (x + 4)(3x - 2)\left(\frac{2}{3x - 2}\right)$}{$\lcm = (x + 4)(3x - 2)$.}
		\phxeqs{$x \not\in \left\{-4, \frac{2}{3}\right\}$}{$3(3x - 2) = 2(x + 4)$}{}
		\phxeqs{$x \not\in \left\{-4, \frac{2}{3}\right\}$}{$9x - 6 = 2x + 8$}{}
		\phxeqs{$x \not\in \left\{-4, \frac{2}{3}\right\}$}{$9x - 2x = 8 + 6$}{}
		\phxeqs{$x \not\in \left\{-4, \frac{2}{3}\right\}$}{$7x = 14$}{}
		\phxeqf{}{$x = 2$}{}
	}

\subsection{} \[\frac{x + 4}{x + 2} = \frac{2x - 3}{x - 5}\]
	\phxproof{
		\phxeqb{}{$(x + 2)(x - 5)\left(\frac{x + 4}{x + 2}\right) = (x + 2)(x - 5)\left(\frac{2x - 3}{x - 5}\right)$}{$\lcm = (x + 2)(x - 5)$.}
		\phxeqs{$x \not\in \{-2, 5\}$}{$(x + 4)(x - 5) = (2x - 3)(x + 2)$}{}
		\phxeqs{$x \not\in \{-2, 5\}$}{$x^2 - x - 20 = 2x^2 + x - 6$}{}
		\phxeqs{$x \not\in \{-2, 5\}$}{$x^2 - 2x^2 - x - x - 20 + 6 = 0$}{Rewrite in standard form.}
		\phxeqs{$x \not\in \{-2, 5\}$}{$-x^2 - 2x - 14 = 0$}{}
		\phxeqs{$x \not\in \{-2, 5\}$}{$x^2 + 2x + 14 = 0$}{}
		\phxeqs{$x \not\in \{-2, 5\}$}{$x = \frac{-2 \pm \sqrt{2^2 - 4(1)(14)}}{2(1)}$}{Use the quadratic formula.}
		\phxeqs{$x \not\in \{-2, 5\}$}{$x = \frac{-2 \pm \sqrt{4 - 56}}{2}$}{}
		\phxeqs{$x \not\in \{-2, 5\}$}{$x = \frac{-2 \pm \sqrt{-52}}{2}$}{}
		\phxeqs{$x \not\in \{-2, 5\}$}{$x = \frac{-2 \pm i\sqrt{4}\sqrt{13}}{2}$}{}
		\phxeqs{$x \not\in \{-2, 5\}$}{$x = \frac{-2 \pm 2i\sqrt{13}}{2}$}{}
		\phxeqf{}{$-1 \pm i\sqrt{13}$}{}
	}

\subsection{} \[\frac{4}{z - 2} = \frac{z}{6}\]
	\phxproof{
		\phxeqb{}{$6(z - 2)\left(\frac{4}{z - 2}\right) = 6(z - 2)\left(\frac{z}{6}\right)$}{$\lcm = 6(z - 2).$}
		\phxeqs{$z \neq 2$}{$6(4) = z(z - 2)$}{}
		\phxeqs{$z \neq 2$}{$24 = z^2 - 2z$}{}
		\phxeqs{$z \neq 2$}{$z^2 - 2z - 24 = 0$}{Rewrite in standard form.}
		\phxeqs{$z \neq 2$}{$(z - 6)(z + 4) = 0$}{}
		\phxeqf{}{$x \in \{6, -4\}$}{}
	}

\subsection{} \[\frac{1}{2x + 5} = \frac{4}{2x - 1}\]
	\phxproof{
		\phxeqb{}{$(2x + 5)(2x - 1)\left(\frac{1}{2x + 5}\right) = (2x + 5)(2x - 1)\left(\frac{4}{2x - 1}\right)$}{$\lcm = (2x + 5)(2x - 1)$.}
		\phxeqs{$x \not\in \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}{$2x - 1 = 4(2x + 5)$}{}
		\phxeqs{$x \not\in \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}{$2x - 1 = 8x + 20$}{}
		\phxeqs{$x \not\in \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}{$2x - 8x = 20 + 1$}{}
		\phxeqs{$x \not\in \left\{-\frac{5}{2}, \frac{1}{2}\right\}$}{$-6x = 21$}{}
		\phxeqf{}{$x = -\frac{7}{2}$}{}
	}

\subsection{} \[\frac{2}{3x} + \frac{4}{5x} = \frac{11}{15}\]
	\phxproof{
		\phxeqb{}{$15x\left(\frac{2}{3x} + \frac{4}{5x}\right) = 15x\left(\frac{11}{15}\right)$}{$\lcm = 15x$.}
		\phxeqs{$x \neq 0$}{$10 + 12 = 11x$}{}
		\phxeqs{$x \neq 0$}{$22 = 11x$}{}
		\phxeqf{}{$x = 2$}{}
	}

\subsection{} \[\frac{5}{x + 3} = \frac{x + 8}{x + 3} - 3\]
	\phxproof{
		\phxeqb{$x \neq -3$}{$5 = x + 8 - 3(x + 3)$}{Eliminate denominator.}
		\phxeqs{$x \neq -3$}{$5 = x + 8 - 3x - 9$}{}
		\phxeqs{$x \neq -3$}{$5 = -2x - 1$}{}
		\phxeqs{$x \neq -3$}{$2x = -1 - 5$}{}
		\phxeqs{$x \neq -3$}{$2x = -6$}{}
		\phxeqs{$x \neq -3$}{$x = -3$}{}
		\phxeqf{}{$x \in \emptyset$}{}
	}

\subsection{} \[\frac{3x}{x + 2} = \frac{x + 4}{x}\]
	\phxproof{
		\phxeqb{}{$x(x + 2)\left(\frac{3x}{x + 2}\right) = x(x + 2)\left(\frac{x + 4}{x}\right)$}{$\lcm = x(x + 2)$.}
		\phxeqs{$x \not\in \{0, -2\}$}{$3x^2 = (x + 2)(x + 4)$}{}
		\phxeqs{$x \not\in \{0, -2\}$}{$3x^2 = x^2 + 6x + 8$}{}
		\phxeqs{$x \not\in \{0, -2\}$}{$3x^2 - x^2 - 6x - 8 = 0$}{Rewrite in standard form.}
		\phxeqs{$x \not\in \{0, -2\}$}{$2x^2 - 6x - 8 = 0$}{}
		\phxeqs{$x \not\in \{0, -2\}$}{$x^2 - 3x - 4 = 0$}{}
		\phxeqs{$x \not\in \{0, -2\}$}{$(x - 4)(x + 1) = 0$}{Factor by grouping.}
		\phxeqf{}{$x \in \{4, -1\}$}{}
	}

\subsection{} \[\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} = \frac{3x - 8}{x^2 - 3x - 10}\]
	\phxproof{
		\phxeqb{}{$\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} = \frac{3x - 8}{(x - 5)(x + 2)}$}{Factor by grouping.}
		\phxeqs{}{$(x - 5)(x + 2)\left(\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5}\right) = (x - 5)(x + 2)\left(\frac{3x - 8}{(x - 5)(x + 2)}\right)$}{$\lcm = (x - 5)(x + 2)$.}
		\phxeqs{$x \not\in \{5, -2\}$}{$(x + 4)(x - 5) - (2x - 3)(x + 2) = 3x - 8$}{}
		\phxeqs{$x \not\in \{5, -2\}$}{$x^2 - x - 20 - (2x^2 + x - 6) = 3x - 8$}{}
		\phxeqs{$x \not\in \{5, -2\}$}{$x^2 - x - 20 - 2x^2 - x + 6 = 3x - 8$}{}
		\phxeqs{$x \not\in \{5, -2\}$}{$x^2 - 2x^2 - x - x - 20 + 6 = 3x - 8$}{}
		\phxeqs{$x \not\in \{5, -2\}$}{$-x^2 - 2x - 14 = 3x - 8$}{}
		\phxeqs{$x \not\in \{5, -2\}$}{$-x^2 - 2x - 3x - 14 + 8 = 0$}{Rewrite in standard form.}
		\phxeqs{$x \not\in \{5, -2\}$}{$-x^2 - 5x - 6 = 0$}{}
		\phxeqs{$x \not\in \{5, -2\}$}{$x^2 + 5x + 6 = 0$}{}
		\phxeqs{$x \not\in \{5, -2\}$}{$(x + 2)(x + 3) = 0$}{Factor by grouping.}
		\phxeqs{$x \not\in \{5, -2\}$}{$x \in \{-2, -3\}$}{}
		\phxeqf{}{$x = -3$}{}
	}
