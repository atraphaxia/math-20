\subsection{} \[\frac{x + 17}{x^2 - 6x + 8} + \frac{x - 2}{x - 4} = \frac{x - 4}{x - 2}\]
	\phxproof{
		\phxeqb{}{$\frac{x + 17}{(x - 2)(x - 4)} + \frac{x - 2}{x - 4} = \frac{x - 4}{x - 2}$}{Factor by grouping.}
		\phxeqs{}{$(x - 2)(x - 4)(\frac{x + 17}{(x - 2)(x - 4)} + \frac{x - 2}{x - 4}) = (x - 2)(x - 4)(\frac{x - 4}{x - 2})$}{$\lcm = (x - 2)(x - 4)$}
		\phxeqs{$x \notin \{2, 4\}$}{$x + 17 + {(x - 2)}^2 = {(x - 4)}^2$}{}
		\phxeqs{$x \notin \{2, 4\}$}{$x + 17 + x^2 - 4x + 4 = x^2 - 8x + 16$}{}
		\phxeqs{$x \notin \{2, 4\}$}{$\cancel{x^2} - 4x + x + 4 + 17 = \cancel{x^2} - 8x + 16$}{}
		\phxeqs{$x \notin \{2, 4\}$}{$-3x + 21 = -8x + 16$}{}
		\phxeqs{$x \notin \{2, 4\}$}{$-3x + 8x = 16 - 21$}{}
		\phxeqs{$x \notin \{2, 4\}$}{$5x = -5$}{}
		\phxeqf{}{$x = -1$}{}
	}

\subsection{} \[\frac{w}{3w^2 - 8w + 4} = \frac{w + 2}{3w^2 + w - 2}\]
	\phxproof{
		\phxeqb{}{$\frac{w}{3w^2 - 6w - 2w + 4} = \frac{w + 2}{3w^2 + 3w - 2w - 2}$}{Factor by grouping.}
		\phxeqs{}{$\frac{w}{(3w - 2)(w - 2)} = \frac{w + 2}{(3w - 2)(w + 1)}$}{}
		\phxeqs{}{\scalebox{0.6}{$(3w - 2)(w - 2)(w + 1)(\frac{w}{(3w - 2)(w - 2)}) = (3w - 2)(w - 2)(w + 1)(\frac{w + 2}{(3w - 2)(w + 1)})$}}{$\lcm = (3w - 2)(w - 2)(w + 1)$}
		\phxeqs{$w \notin \{\frac{2}{3}, 2, -1\}$}{$w(w + 1) = (w + 2)(w - 2)$}{}
		\phxeqs{$w \notin \{\frac{2}{3}, 2, -1\}$}{$\cancel{w^2} + w = \cancel{w^2} - 4$}{Expand using difference of two squares.}
		\phxeqf{}{$w = -4$}{}
	}

\subsection{} \[\frac{3}{x + 4} = \frac{2}{3x - 2}\]
	\phxproof{
		\phxeqb{}{$(x + 4)(3x - 2)(\frac{3}{x + 4}) = (x + 4)(3x - 2)(\frac{2}{3x - 2})$}{$\lcm = (x + 4)(3x - 2)$}
		\phxeqs{$x \notin \{-4, \frac{2}{3}\}$}{$3(3x - 2) = 2(x + 4)$}{}
		\phxeqs{$x \notin \{-4, \frac{2}{3}\}$}{$9x - 6 = 2x + 8$}{}
		\phxeqs{$x \notin \{-4, \frac{2}{3}\}$}{$9x - 2x = 8 + 6$}{}
		\phxeqs{$x \notin \{-4, \frac{2}{3}\}$}{$7x = 14$}{}
		\phxeqf{}{$x = 2$}{}
	}

\subsection{} \[\frac{x + 4}{x + 2} = \frac{2x - 3}{x - 5}\]
	\phxproof{
		\phxeqb{}{$(x + 2)(x - 5)(\frac{x + 4}{x + 2}) = (x + 2)(x - 5)(\frac{2x - 3}{x - 5})$}{$\lcm = (x + 2)(x - 5)$}
		\phxeqs{$x \notin \{-2, 5\}$}{$(x + 4)(x - 5) = (2x - 3)(x + 2)$}{}
		\phxeqs{$x \notin \{-2, 5\}$}{$x^2 - x - 20 = 2x^2 + x - 6$}{}
		\phxeqs{$x \notin \{-2, 5\}$}{$x^2 - 2x^2 - x - x - 20 + 6 = 0$}{Rewrite in standard form.}
		\phxeqs{$x \notin \{-2, 5\}$}{$-x^2 - 2x - 14 = 0$}{}
		\phxeqs{$x \notin \{-2, 5\}$}{$x^2 + 2x + 14 = 0$}{}
		\phxeqs{$x \notin \{-2, 5\}$}{$x = \frac{-2 \pm \sqrt{2^2 - 4(1)(14)}}{2(1)}$}{Use the quadratic formula.}
		\phxeqs{$x \notin \{-2, 5\}$}{$x = \frac{-2 \pm \sqrt{4 - 56}}{2}$}{}
		\phxeqs{$x \notin \{-2, 5\}$}{$x = \frac{-2 \pm \sqrt{-52}}{2}$}{}
		\phxeqs{$x \notin \{-2, 5\}$}{$x = \frac{-2 \pm i\sqrt{4}\sqrt{13}}{2}$}{}
		\phxeqs{$x \notin \{-2, 5\}$}{$x = \frac{-2 \pm 2i\sqrt{13}}{2}$}{}
		\phxeqf{}{$-1 \pm i\sqrt{13}$}{}
	}

\subsection{} \[\frac{4}{z - 2} = \frac{z}{6}\]
	\phxproof{
		\phxeqb{}{$6(z - 2)(\frac{4}{z - 2}) = 6(z - 2)(\frac{z}{6})$}{$\lcm = 6(z - 2)$}
		\phxeqs{$z \ne 2$}{$6(4) = z(z - 2)$}{}
		\phxeqs{$z \ne 2$}{$24 = z^2 - 2z$}{}
		\phxeqs{$z \ne 2$}{$z^2 - 2z - 24 = 0$}{Rewrite in standard form.}
		\phxeqs{$z \ne 2$}{$(z - 6)(z + 4) = 0$}{}
		\phxeqf{}{$x \in \{6, -4\}$}{}
	}

\subsection{} \[\frac{1}{2x + 5} = \frac{4}{2x - 1}\]
	\phxproof{
		\phxeqb{}{$(2x + 5)(2x - 1)(\frac{1}{2x + 5}) = (2x + 5)(2x - 1)(\frac{4}{2x - 1})$}{$\lcm = (2x + 5)(2x - 1)$}
		\phxeqs{$x \notin \{-\frac{5}{2}, \frac{1}{2}\}$}{$2x - 1 = 4(2x + 5)$}{}
		\phxeqs{$x \notin \{-\frac{5}{2}, \frac{1}{2}\}$}{$2x - 1 = 8x + 20$}{}
		\phxeqs{$x \notin \{-\frac{5}{2}, \frac{1}{2}\}$}{$2x - 8x = 20 + 1$}{}
		\phxeqs{$x \notin \{-\frac{5}{2}, \frac{1}{2}\}$}{$-6x = 21$}{}
		\phxeqf{}{$x = -\frac{7}{2}$}{}
	}

\subsection{} \[\frac{2}{3x} + \frac{4}{5x} = \frac{11}{15}\]
	\phxproof{
		\phxeqb{}{$15x(\frac{2}{3x} + \frac{4}{5x}) = 15x(\frac{11}{15})$}{$\lcm = 15x$}
		\phxeqs{$x \ne 0$}{$10 + 12 = 11x$}{}
		\phxeqs{$x \ne 0$}{$22 = 11x$}{}
		\phxeqf{}{$x = 2$}{}
	}

\subsection{} \[\frac{5}{x + 3} = \frac{x + 8}{x + 3} - 3\]
	\phxproof{
		\phxeqb{$x \ne -3$}{$5 = x + 8 - 3(x + 3)$}{Eliminate denominator.}
		\phxeqs{$x \ne -3$}{$5 = x + 8 - 3x - 9$}{}
		\phxeqs{$x \ne -3$}{$5 = -2x - 1$}{}
		\phxeqs{$x \ne -3$}{$2x = -1 - 5$}{}
		\phxeqs{$x \ne -3$}{$2x = -6$}{}
		\phxeqs{$x \ne -3$}{$x = -3$}{}
		\phxeqf{}{$x \in \emptyset$}{}
	}

\subsection{} \[\frac{3x}{x + 2} = \frac{x + 4}{x}\]
	\phxproof{
		\phxeqb{}{$x(x + 2)(\frac{3x}{x + 2}) = x(x + 2)(\frac{x + 4}{x})$}{$\lcm = x(x + 2)$}
		\phxeqs{$x \notin \{0, -2\}$}{$3x^2 = (x + 2)(x + 4)$}{}
		\phxeqs{$x \notin \{0, -2\}$}{$3x^2 = x^2 + 6x + 8$}{}
		\phxeqs{$x \notin \{0, -2\}$}{$3x^2 - x^2 - 6x - 8 = 0$}{Rewrite in standard form.}
		\phxeqs{$x \notin \{0, -2\}$}{$2x^2 - 6x - 8 = 0$}{}
		\phxeqs{$x \notin \{0, -2\}$}{$x^2 - 3x - 4 = 0$}{}
		\phxeqs{$x \notin \{0, -2\}$}{$(x - 4)(x + 1) = 0$}{Factor by grouping.}
		\phxeqf{}{$x \in \{4, -1\}$}{}
	}

\subsection{} \[\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} = \frac{3x - 8}{x^2 - 3x - 10}\]
	\phxproof{
		\phxeqb{}{$\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5} = \frac{3x - 8}{(x - 5)(x + 2)}$}{Factor by grouping.}
		\phxeqs{}{$(x - 5)(x + 2)(\frac{x + 4}{x + 2} - \frac{2x - 3}{x - 5}) = (x - 5)(x + 2)(\frac{3x - 8}{(x - 5)(x + 2)})$}{$\lcm = (x - 5)(x + 2)$}
		\phxeqs{$x \notin \{5, -2\}$}{$(x + 4)(x - 5) - (2x - 3)(x + 2) = 3x - 8$}{}
		\phxeqs{$x \notin \{5, -2\}$}{$x^2 - x - 20 - (2x^2 + x - 6) = 3x - 8$}{}
		\phxeqs{$x \notin \{5, -2\}$}{$x^2 - x - 20 - 2x^2 - x + 6 = 3x - 8$}{}
		\phxeqs{$x \notin \{5, -2\}$}{$x^2 - 2x^2 - x - x - 20 + 6 = 3x - 8$}{}
		\phxeqs{$x \notin \{5, -2\}$}{$-x^2 - 2x - 14 = 3x - 8$}{}
		\phxeqs{$x \notin \{5, -2\}$}{$-x^2 - 2x - 3x - 14 + 8 = 0$}{Rewrite in standard form.}
		\phxeqs{$x \notin \{5, -2\}$}{$-x^2 - 5x - 6 = 0$}{}
		\phxeqs{$x \notin \{5, -2\}$}{$x^2 + 5x + 6 = 0$}{}
		\phxeqs{$x \notin \{5, -2\}$}{$(x + 2)(x + 3) = 0$}{Factor by grouping.}
		\phxeqs{$x \notin \{5, -2\}$}{$x \in \{-2, -3\}$}{}
		\phxeqf{}{$x = -3$}{}
	}
